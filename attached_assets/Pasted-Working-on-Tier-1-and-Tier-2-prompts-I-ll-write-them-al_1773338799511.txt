```ts
const stripe = new Stripe(Deno.env.get('STRIPE_SECRET_KEY')!);

  return new Response(

      clientSecret: verificationSession.client_secret 
    }),

  );
});
```

```ts
const COMPATIBLE_PAIRS: Record<string, string[]> = {
  'q1_alone': ['q1_alone', 'q1_music'],
  'q1_social': ['q1_social', 'q1_kitchen'],
  'q2_clean': ['q2_clean'],
  'q2_chill': ['q2_chill', 'q2_out'],
  'q2_social': ['q2_social'],
  'q3_spotless': ['q3_spotless', 'q3_tidy'],
  'q3_relaxed': ['q3_relaxed', 'q3_doesnt_matter'],
  'q4_advance': ['q4_advance', 'q4_headsup'],
  'q4_open': ['q4_open', 'q4_love'],
};

const calculatePersonalityScore = (answers1, answers2): number => {
  let matches = 0;
  let total = 0;
  Object.keys(answers1).forEach(q => {
    const a1 = `${q}_${answers1[q]}`;
    const a2 = `${q}_${answers2[q]}`;
    const compatible = COMPATIBLE_PAIRS[a1]?.includes(a2);
    if (compatible) matches++;
    total++;
  });
  return total > 0 ? (matches / total) * 100 : 50;
};
```
