const getPhotoUrl = (seed) => `https://randomuser.me/api/portraits/${seed % 2 === 0 ? 'women' : 'men'}/${(seed % 70) + 1}.jpg`;

function randomBetween(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

function pick(arr) {
  return arr[Math.floor(Math.random() * arr.length)];
}

// ─── MAIN SEED FUNCTION ───────────────────────────────────────────────────────

async function seedUsers() {
  console.log(`\n🌱 Rhome Seed Script — creating ${SEED_COUNT} fake renters in ${SEED_CITY}...\n`);

  let created = 0;
  let skipped = 0;
  let failed = 0;

  for (let i = 0; i < SEED_COUNT; i++) {
    const { email, password, user: userData, profile: profileData } = generateUser(i);

    try {
      // Step 1: Create auth user
      const { data: authData, error: authError } = await supabase.auth.admin.createUser({
        email,
        password,
        email_confirm: true, // skip email verification
      });

      if (authError) {
        if (authError.message.includes('already been registered')) {
          console.log(`  ⏭  Skipped ${email} (already exists)`);
          skipped++;
          continue;
        }
        throw authError;
      }

      const userId = authData.user.id;

      // Step 2: Insert into users table
      const { error: userError } = await supabase
        .from('users')
        .upsert({ id: userId, email, ...userData });

      if (userError) throw userError;

      // Step 3: Insert into profiles table
      const { error: profileError } = await supabase
        .from('profiles')
        .upsert({ user_id: userId, ...profileData });

      if (profileError) throw profileError;

      console.log(`  ✅ Created ${userData.full_name} (${email})`);
      created++;

      // Small delay to avoid rate limiting
      await new Promise(r => setTimeout(r, 150));

    } catch (err) {
      console.error(`  ❌ Failed ${email}:`, err.message);
      failed++;
    }
  }

  console.log(`\n─────────────────────────────────`);
  console.log(`✅ Created:  ${created}`);
  console.log(`⏭  Skipped:  ${skipped}`);
  console.log(`❌ Failed:   ${failed}`);
  console.log(`─────────────────────────────────`);
  console.log(`\nDone! Open the app and start swiping.\n`);
  console.log(`All seed accounts use password: ${SEED_PASSWORD}`);
  console.log(`Email format: seed.user.1@rhome-test.com → seed.user.${SEED_COUNT}@rhome-test.com\n`);
}

seedUsers().catch(console.error);
