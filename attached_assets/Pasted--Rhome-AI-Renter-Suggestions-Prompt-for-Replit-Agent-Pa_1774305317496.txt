#### Fetching suggestions:
Call a new utility function: getSuggestedGroupMembers(groupId, currentUserId)

Create this function in utils/groupSuggestions.ts:

  import { supabase } from '../lib/supabase';
  import { calculateGroupCompatibilityScore } from './transitMatching';

  export async function getSuggestedGroupMembers(
    groupId: string,
    currentUserId: string,
    limit = 3
  ): Promise<Array<{ profile: any; groupScore: number; reason: string }>> {
    // 1. Fetch the group and its current members
    const { data: group } = await supabase
      .from('groups')
      .select('*, group_members(user_id)')
      .eq('id', groupId)
      .single();

    if (!group) return [];
    const memberIds: string[] = group.group_members.map((m: any) => m.user_id);
    const desiredBedrooms = group.desired_bedrooms ?? 2;

    if (memberIds.length >= desiredBedrooms) return []; // group is full

    // 2. Fetch member profiles (with apartmentPrefs)
    const { data: memberUsers } = await supabase
      .from('users')
      .select('*, profile:profiles(*)')
      .in('id', memberIds);

    // 3. Fetch candidate renters NOT in the group, NOT already invited
    const { data: invited } = await supabase
      .from('group_invitations')
      .select('invitee_id')
      .eq('group_id', groupId);
    const invitedIds = (invited ?? []).map((i: any) => i.invitee_id);
    const excludeIds = [...memberIds, ...invitedIds, currentUserId];

    const { data: candidates } = await supabase
      .from('users')
      .select('*, profile:profiles(*)')
      .eq('role', 'renter')
      .eq('onboarding_step', 'complete')
      .not('id', 'in', `(${excludeIds.join(',')})`)
      .limit(30);

    if (!candidates || candidates.length === 0) return [];

    // 4. For each candidate, score compatibility against ALL current group members
    const scored = candidates.map(candidate => {
      const candidateProfile = mapToRoommateProfile(candidate);
      let totalScore = 0;
      let conflicts: string[] = [];

      for (const member of (memberUsers ?? [])) {
        const memberProfile = mapToRoommateProfile(member);
        const result = calculateGroupCompatibilityScore(memberProfile, candidateProfile);
        totalScore += result.total;
        conflicts.push(...result.conflicts);
      }

      const avgScore = memberUsers && memberUsers.length > 0
        ? Math.round(totalScore / memberUsers.length)
        : 50;

      let reason = '';
      if (avgScore >= 85) reason = 'Strong lifestyle and budget match with your whole group';
      else if (avgScore >= 70) reason = 'Good fit — similar schedule and budget range';
      else if (avgScore >= 55) reason = 'Compatible on most factors, check move-in date';
      else reason = 'Some differences — worth a conversation';

      return { profile: candidate, groupScore: avgScore, reason };
    });

    return scored
      .filter(s => s.groupScore >= 55)
      .sort((a, b) => b.groupScore - a.groupScore)
      .slice(0, limit);
  }

  // Helper: map Supabase user+profile row to RoommateProfile shape
  function mapToRoommateProfile(user: any): any {
    const p = user.profile ?? {};
    return {
      id: user.id,
      name: user.full_name,
      age: user.age,
      lifestyle: {
        cleanliness: p.cleanliness,
        workSchedule: p.sleep_schedule,
        smoking: p.smoking,
        pets: p.pets,
        drinking: p.drinking,
        guests: p.guests,
        noiseLevel: p.noise_tolerance,
      },
      budget: p.budget_max,
      preferences: { moveInDate: p.move_in_date },
      apartmentPrefs: p.apartment_prefs_complete ? {
        desiredBedrooms: p.desired_bedrooms,
        budgetPerPersonMin: p.budget_per_person_min,
        budgetPerPersonMax: p.budget_per_person_max,
        preferredTrains: p.preferred_trains ?? [],
        preferredNeighborhoods: p.preferred_neighborhoods ?? [],
        amenityMustHaves: p.amenity_must_haves ?? [],
        moveInDate: p.move_in_date,
        locationFlexible: p.location_flexible,
        wfh: p.wfh,
        apartmentPrefsComplete: true,
      } : undefined,
    };
  }

### PART 2: "Best Match Today" AI Card in RoommatesScreen Swipe Feed

File: screens/renter/RoommatesScreen.tsx

Add a pinned card at the top of the swipe deck (before the first regular card)
that highlights the user's single best match for the day.

#### Logic:
- Query match_scores where user_id_1 = userId OR user_id_2 = userId
- Order by score DESC, limit 1
- Only show if score >= 80
- Cache with a 24-hour TTL using AsyncStorage key '@rhome/best_match_today_{userId}'
- Don't show the same person twice in the same day
- Only show for plus/elite users; free users see a teaser

#### Create utility: utils/bestMatchToday.ts

  import { supabase } from '../lib/supabase';
  import AsyncStorage from '@react-native-async-storage/async-storage';

  const TTL_MS = 24 * 60 * 60 * 1000;

  export async function getBestMatchToday(userId: string): Promise<{
    profile: any; score: number; reason: string;
  } | null> {
    const cacheKey = `@rhome/best_match_today_${userId}`;
    const cached = await AsyncStorage.getItem(cacheKey);
    if (cached) {
      const parsed = JSON.parse(cached);
      if (Date.now() - parsed.cachedAt < TTL_MS) return parsed.data;
    }

    const { data: scoreRows } = await supabase
      .from('match_scores')
      .select('user_id_1, user_id_2, score, breakdown')
      .or(`user_id_1.eq.${userId},user_id_2.eq.${userId}`)
      .gte('score', 80)
      .order('score', { ascending: false })
      .limit(1);

    if (!scoreRows || scoreRows.length === 0) return null;

    const row = scoreRows[0];
    const otherId = row.user_id_1 === userId ? row.user_id_2 : row.user_id_1;

    const { data: otherUser } = await supabase
      .from('users')
      .select('*, profile:profiles(*)')
      .eq('id', otherId)
      .single();

    if (!otherUser) return null;

    let reason = '';
    const b = row.breakdown ?? {};
    if (b.sleep_schedule >= 20 && b.cleanliness >= 15) {
      reason = 'Same sleep schedule and cleanliness level';
    } else if (b.budget >= 20) {
      reason = 'Budget and lifestyle are a strong match';
    } else {
      reason = 'High compatibility across multiple factors';
    }

    const result = { profile: otherUser, score: row.score, reason };
    await AsyncStorage.setItem(cacheKey, JSON.stringify({ data: result, cachedAt: Date.now() }));
    return result;
  }
