```ts
});
```

**PROMPT 2 — Environment-Safe Mock Data**

Do not delete or modify any mock data files. Add a single environment check so mock data is blocked in production but works freely in development.

Create `utils/dataUtils.ts`:
```ts
import { isDev } from './envUtils';

export const getMockFallback = <T>(
  mockData: T,
  realData: T | null | undefined
): T | null => {
  // Real data always takes priority
  if (realData !== null && realData !== undefined) return realData;
  // In development, fall back to mock data freely
  if (isDev) return mockData;
  // In production, return null and show empty state
  return null;
};
```

Find every place in the codebase where mock data is used as a fallback using patterns like `?? mockData`, `|| mockData`, or `= mockData` and wrap with `getMockFallback`:

```ts
// Before
const profiles = data ?? mockProfiles;

// After
import { getMockFallback } from '../../utils/dataUtils';
const profiles = getMockFallback(mockProfiles, data);
```

This is the ONLY change needed. All mock data files stay intact. All mock imports stay intact. The only difference is production returns null instead of mock data when real data is unavailable.

```ts
import Stripe from 'https://esm.sh/stripe@12.0.0?target=deno';

const stripe = new Stripe(Deno.env.get('STRIPE_SECRET_KEY')!);

Deno.serve(async (req) => {
  const { userId } = await req.json();
  const session = await stripe.identity.verificationSessions.create({

      }
    }
  });
  return new Response(
    JSON.stringify({
      sessionId: session.id,
      clientSecret: session.client_secret
    }),
    { headers: { 'Content-Type': 'application/json' } }
  );
});
```

```ts
      `https://verify.stripe.com/start#${data.clientSecret}`,
      'roomdr://verification-complete'
    );
    if (result.type === 'success') {
      await pollVerificationStatus();
    }
  } catch (error) {
    Alert.alert('Verification failed', error.message);
  } finally {
    setLoading(false);
  }

const pollVerificationStatus = async () => {
  let attempts = 0;
  const interval = setInterval(async () => {
    const { data } = await supabase
      .from('users')
      .select('identity_verified')
      .eq('id', user.id)
      .single();
    if (data?.identity_verified || attempts > 10) {
      clearInterval(interval);
      if (data?.identity_verified) navigation.navigate('VerificationSuccess');
    }
    attempts++;
  }, 3000);
};
```

```ts
const COMPATIBLE_ANSWERS: Record<string, string[]> = {
  q1_alone: ['q1_alone', 'q1_music'],
  q1_social: ['q1_social', 'q1_kitchen'],
  q2_clean: ['q2_clean'],
  q2_chill: ['q2_chill', 'q2_out'],
  q2_social: ['q2_social'],
  q3_spotless: ['q3_spotless', 'q3_tidy'],
  q3_relaxed: ['q3_relaxed', 'q3_doesnt_matter'],
  q4_advance: ['q4_advance', 'q4_headsup'],
  q4_open: ['q4_open', 'q4_love'],
  q5_text: ['q5_text', 'q5_flow'],
  q5_direct: ['q5_direct', 'q5_meeting'],
};

const calculatePersonalityScore = (
  answers1: Record<string, string>,
  answers2: Record<string, string>
): number => {
  if (!answers1 || !answers2) return 50;
  let matches = 0;
  let total = 0;
  Object.keys(answers1).forEach(q => {
    const key = `${q}_${answers1[q]}`;
    const val2 = `${q}_${answers2[q]}`;
    if (COMPATIBLE_ANSWERS[key]?.includes(val2)) matches++;
    total++;
  });
  return total > 0 ? (matches / total) * 100 : 50;
};

totalScore += calculatePersonalityScore(
  user1.profile?.personality_answers,
  user2.profile?.personality_answers
) * 0.15;
```
