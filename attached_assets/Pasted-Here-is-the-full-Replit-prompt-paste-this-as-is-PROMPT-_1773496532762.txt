interface AIMemory {
  // Swipe behavior
  totalSwipes: number;
  rightSwipes: number;
  leftSwipes: number;
  skipReasons: string[]; // e.g. ['too far', 'night shift', 'smoker']
  avgMatchScore: number;
  matchScoreTrend: 'improving' | 'declining' | 'stable';

  // Filter behavior
  skippedDueToDistance: number;
  skippedDueToSchedule: number;
  skippedDueToBudget: number;
  autoTightenedFilters: string[]; // filters the AI has already suggested tightening

  // Chat behavior
  totalConversations: number;
  coldConversations: string[]; // chatIds with no reply in 5+ days
  avgResponseRatePercent: number;
  lastActiveByChat: Record<string, string>; // chatId -> ISO timestamp

  // Profile
  profileCompletionPercent: number;
  missingFields: string[];
  matchesBlockedByMissingFields: number; // how many top matches filter by fields you haven't filled

  // Move-in urgency
  moveInDate: string | null; // ISO date from profile
  daysUntilMoveIn: number | null;

  // Personality refinement
  refinementQuestionsAnswered: number;
  lastRefinementTimestamp: string | null;

  // Session
  lastUpdated: string; // ISO timestamp
}

Create these helper functions in utils/aiMemory.ts:
- getAIMemory(): Promise<AIMemory> — loads from AsyncStorage
- updateAIMemory(patch: Partial<AIMemory>): Promise<void> — merges and saves
- resetAIMemory(): Promise<void> — clears all memory (dev tool only)
- computeSkipReasons(swipeHistory: any[]): string[] — infers reasons from rejected profiles

Update memory after every swipe in RoommatesScreen.tsx by calling updateAIMemory().
Update memory after every message sent in the chat screen.
