Set Up Supabase Backend for Roomdr
Integrate Supabase as the full backend for this React Native / Expo app. Replace all mocked data, fake auth, and local state with real Supabase services. Follow every step completely before moving on to the next.

STEP 1 — Install Dependencies
bashnpx expo install @supabase/supabase-js
npx expo install @react-native-async-storage/async-storage
npx expo install react-native-url-polyfill

STEP 2 — Create Supabase Client
Create lib/supabase.ts:
tsimport 'react-native-url-polyfill/auto';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { createClient } from '@supabase/supabase-js';

const supabaseUrl = process.env.EXPO_PUBLIC_SUPABASE_URL!;
const supabaseAnonKey = process.env.EXPO_PUBLIC_SUPABASE_ANON_KEY!;

export const supabase = createClient(supabaseUrl, supabaseAnonKey, {
  auth: {
    storage: AsyncStorage,
    autoRefreshToken: true,
    persistSession: true,
    detectSessionInUrl: false,
  },
});
```

Create a `.env` file in the project root:
```
EXPO_PUBLIC_SUPABASE_URL=your_supabase_project_url
EXPO_PUBLIC_SUPABASE_ANON_KEY=your_supabase_anon_key
Add .env to .gitignore if it isn't already there.

STEP 3 — Database Schema
Create the following tables in Supabase. Write these as SQL migration files in a new folder supabase/migrations/:
users table
sqlcreate table public.users (
  id uuid references auth.users(id) primary key,
  email text unique not null,
  role text check (role in ('renter', 'host')) not null default 'renter',
  full_name text,
  avatar_url text,
  bio text,
  age integer,
  gender text,
  location text,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);
profiles table (renter profile details)
sqlcreate table public.profiles (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references public.users(id) on delete cascade unique,
  budget_min integer,
  budget_max integer,
  move_in_date date,
  lease_duration text,
  work_location text check (work_location in ('wfh_fulltime', 'hybrid', 'office_fulltime', 'irregular')),
  cleanliness integer check (cleanliness between 1 and 5),
  noise_level integer check (noise_level between 1 and 5),
  sleep_schedule text,
  pets text,
  smoking boolean default false,
  photos text[],
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);
listings table (host listings)
sqlcreate table public.listings (
  id uuid primary key default gen_random_uuid(),
  host_id uuid references public.users(id) on delete cascade,
  title text not null,
  description text,
  rent integer not null,
  bedrooms integer,
  bathrooms integer,
  address text,
  city text,
  state text,
  photos text[],
  available_date date,
  is_active boolean default true,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);
matches table
sqlcreate table public.matches (
  id uuid primary key default gen_random_uuid(),
  user_id_1 uuid references public.users(id) on delete cascade,
  user_id_2 uuid references public.users(id) on delete cascade,
  match_type text check (match_type in ('mutual', 'super_interest', 'cold')) default 'mutual',
  status text check (status in ('pending', 'matched', 'rejected')) default 'pending',
  created_at timestamptz default now(),
  unique(user_id_1, user_id_2)
);
messages table
sqlcreate table public.messages (
  id uuid primary key default gen_random_uuid(),
  match_id uuid references public.matches(id) on delete cascade,
  sender_id uuid references public.users(id) on delete cascade,
  content text not null,
  read boolean default false,
  created_at timestamptz default now()
);
subscriptions table
sqlcreate table public.subscriptions (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references public.users(id) on delete cascade unique,
  plan text check (plan in ('basic', 'plus', 'elite', 'starter', 'pro', 'business')) default 'basic',
  billing_cycle text check (billing_cycle in ('monthly', '3month', 'annual')),

  current_period_end timestamptz,
  cancel_at_period_end boolean default false,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);
boosts table
sqlcreate table public.boosts (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references public.users(id) on delete cascade,
  is_active boolean default true,
  expires_at timestamptz not null,
  duration_hours integer check (duration_hours in (12, 24, 48)),
  created_at timestamptz default now()
);
super_interests table
sqlcreate table public.super_interests (
  id uuid primary key default gen_random_uuid(),

STEP 4 — Row Level Security
Enable RLS on every table and add these policies:
sql-- Users can only read/update their own user record
alter table public.users enable row level security;
create policy "Users can view own profile" on public.users for select using (auth.uid() = id);
create policy "Users can update own profile" on public.users for update using (auth.uid() = id);

-- Profiles visible to all authenticated users (for matching)
alter table public.profiles enable row level security;
create policy "Profiles visible to authenticated users" on public.profiles for select using (auth.role() = 'authenticated');
create policy "Users manage own profile" on public.profiles for all using (auth.uid() = user_id);

-- Listings visible to all authenticated users
alter table public.listings enable row level security;
create policy "Listings visible to authenticated users" on public.listings for select using (auth.role() = 'authenticated');
create policy "Hosts manage own listings" on public.listings for all using (auth.uid() = host_id);

-- Messages only visible to match participants
alter table public.messages enable row level security;
create policy "Match participants can read messages" on public.messages for select using (
  auth.uid() in (
    select user_id_1 from public.matches where id = match_id
    union
    select user_id_2 from public.matches where id = match_id
  )
);
create policy "Match participants can send messages" on public.messages for insert with check (auth.uid() = sender_id);

-- Subscriptions private to user
alter table public.subscriptions enable row level security;
create policy "Users view own subscription" on public.subscriptions for select using (auth.uid() = user_id);

STEP 7 — Realtime Messaging
In ChatScreen.tsx, replace any local message state with a Supabase Realtime subscription:
tsuseEffect(() => {
  const channel = supabase
    .channel(`match-${matchId}`)
    .on('postgres_changes', {
      event: 'INSERT',
      schema: 'public',
      table: 'messages',
      filter: `match_id=eq.${matchId}`
    }, (payload) => {
      setMessages(prev => [...prev, payload.new as Message]);
    })
    .subscribe();

  return () => { supabase.removeChannel(channel); };
}, [matchId]);

STEP 8 — Stripe Webhook Edge Function
Create supabase/functions/stripe-webhook/index.ts:
This Edge Function should:

Receive POST requests from Stripe webhooks
Verify the Stripe webhook signature using STRIPE_WEBHOOK_SECRET
Handle these events:

customer.subscription.created → set subscription status to active
customer.subscription.updated → update plan, billing cycle, period end
customer.subscription.deleted → set status to cancelled
invoice.payment_failed → set status to past_due


Update the subscriptions table accordingly after each event

Add these secrets to Supabase Edge Function environment:

STRIPE_SECRET_KEY
STRIPE_WEBHOOK_SECRET


STEP 9 — Supabase Storage for Photos
Create two storage buckets in Supabase:

profile-photos — for user profile and renter profile images
listing-photos — for host listing images

In EditProfileScreen.tsx and any screen that handles photo uploads:

Replace any local image state with uploads to Supabase Storage
Use supabase.storage.from('profile-photos').upload() to upload
Use supabase.storage.from('profile-photos').getPublicUrl() to get the URL
Store the returned public URL in the profiles.photos array in the database
