# Rhome — Transit-Aware Matching System: Complete Build Prompt

Build the **pre-group apartment preference questionnaire**, **transit-aware matching algorithm**, **AI apartment suggestions**, and **updated agent matchmaker with transit pre-filtering**. This system ensures renters are aligned on location, budget, and commute BEFORE they form a group — and agents can only match renters who already fit their listing's neighborhood and transit access.

---

## The Core Philosophy

**Wrong order (current):**
```
Match on lifestyle → form group → figure out where to live → conflict → group dissolves
```

**Right order (new):**
```
Declare apartment preferences + train lines →
AI finds people who want the same thing →
Match on lifestyle →
Form group already aligned on everything →
AI suggests apartments that work for the whole group
```

```ts
// Find neighborhoods compatible with a set of train lines
export function findCompatibleNeighborhoods(trainLines: string[]): string[] {
  if (!trainLines || trainLines.length === 0) return [];

  return Object.entries(NEIGHBORHOOD_TRAINS)
    .filter(([neighborhood, trains]) =>
      trainLines.every(line =>
        trains.includes(line) ||
        hasReasonableTransfer(line, trains)
      )
    )
    .map(([neighborhood]) => neighborhood);
}

// Allow 1 transfer for flexibility
function hasReasonableTransfer(line: string, neighborhoodTrains: string[]): boolean {
  const transferHubs: Record<string, string[]> = {
    'L': ['G', 'J', 'M', 'Z'],
    'G': ['L', 'F', 'R', 'A', 'C'],
    'A': ['C', 'E', '1', '2', '3'],
    '1': ['2', '3', 'A', 'C'],
    '4': ['5', '6', 'N', 'Q', 'R'],
    'N': ['Q', 'R', 'W', '4', '5', '6'],
  };
  const transfers = transferHubs[line] ?? [];
  return transfers.some(t => neighborhoodTrains.includes(t));
}
```

## Part 3 — Updated Matching Algorithm

Update `utils/matchingAlgorithm.ts` to include transit and apartment preference scoring.

```ts
export interface ApartmentPreferences {
  desiredBedrooms: number;
  budgetPerPersonMin: number;
  budgetPerPersonMax: number;
  preferredTrains: string[];
  preferredNeighborhoods: string[];
  amenityMustHaves: string[];
  moveInDate: string;
  locationFlexible: boolean;
  wfh: boolean;
}

// Full group compatibility score — used for suggesting group matches
export function calculateGroupCompatibilityScore(
  userA: User & { apartmentPrefs: ApartmentPreferences },
  userB: User & { apartmentPrefs: ApartmentPreferences }
): {
  total: number;
  breakdown: {
    lifestyle: number;
    budget: number;
    transit: number;
    moveIn: number;
    amenities: number;
  };
  conflicts: string[];
  sharedNeighborhoods: string[];
} {
  const conflicts: string[] = [];

  // ── 1. LIFESTYLE SCORE (35%) ──────────────────────────
  const lifestyleScore = calculateCompatibility(userA, userB); // existing function

  // ── 2. BUDGET SCORE (30%) ────────────────────────────
  // Do their per-person budgets overlap?
  const prefsA = userA.apartmentPrefs;
  const prefsB = userB.apartmentPrefs;

  const budgetOverlapMin = Math.max(prefsA.budgetPerPersonMin, prefsB.budgetPerPersonMin);
  const budgetOverlapMax = Math.min(prefsA.budgetPerPersonMax, prefsB.budgetPerPersonMax);
  const budgetOverlaps = budgetOverlapMax >= budgetOverlapMin;

  let budgetScore = 0;
  if (budgetOverlaps) {
    const overlapRange = budgetOverlapMax - budgetOverlapMin;
    const totalRange = Math.max(prefsA.budgetPerPersonMax, prefsB.budgetPerPersonMax) -
                       Math.min(prefsA.budgetPerPersonMin, prefsB.budgetPerPersonMin);
    budgetScore = Math.min(100, (overlapRange / totalRange) * 100 + 50);
  } else {
    conflicts.push(`Budget mismatch: $${prefsA.budgetPerPersonMin}–$${prefsA.budgetPerPersonMax} vs $${prefsB.budgetPerPersonMin}–$${prefsB.budgetPerPersonMax}`);
  }

  // ── 3. TRANSIT SCORE (25%) ───────────────────────────
  let transitScore = 0;
  let sharedNeighborhoods: string[] = [];

  if (prefsA.locationFlexible || prefsB.locationFlexible || prefsA.wfh || prefsB.wfh) {
    transitScore = 90; // flexible users are compatible with anyone
  } else {
    const allTrains = [...new Set([...prefsA.preferredTrains, ...prefsB.preferredTrains])];
    sharedNeighborhoods = findCompatibleNeighborhoods(allTrains);

    if (sharedNeighborhoods.length >= 3) transitScore = 100;
    else if (sharedNeighborhoods.length === 2) transitScore = 80;
    else if (sharedNeighborhoods.length === 1) transitScore = 60;
    else {
      transitScore = 0;
      conflicts.push(
        `No neighborhood works for both commutes: ${prefsA.preferredTrains.join('/')} vs ${prefsB.preferredTrains.join('/')}`
      );
    }
  }

  // ── 4. MOVE-IN DATE SCORE (10%) ──────────────────────
  let moveInScore = 100;
  if (prefsA.moveInDate && prefsB.moveInDate) {
    const dateA = new Date(prefsA.moveInDate);
    const dateB = new Date(prefsB.moveInDate);
    const diffDays = Math.abs((dateA.getTime() - dateB.getTime()) / (1000 * 60 * 60 * 24));

    if (diffDays <= 14) moveInScore = 100;
    else if (diffDays <= 30) moveInScore = 80;
    else if (diffDays <= 60) moveInScore = 50;
    else {
      moveInScore = 0;
      conflicts.push(`Move-in dates are ${Math.round(diffDays)} days apart`);
    }
  }

  // ── 5. AMENITY SCORE (bonus, not weighted) ───────────
  const sharedAmenities = prefsA.amenityMustHaves.filter(a =>
    prefsB.amenityMustHaves.includes(a)
  );
  const amenityScore = sharedAmenities.length > 0
    ? Math.min(100, sharedAmenities.length * 25)
    : 50; // no conflict if different preferences

  // ── TOTAL ─────────────────────────────────────────────
  const total = Math.round(
    lifestyleScore * 0.35 +
    budgetScore    * 0.30 +
    transitScore   * 0.25 +
    moveInScore    * 0.10
  );

  return {
    total,
    breakdown: {
      lifestyle: Math.round(lifestyleScore),
      budget:    Math.round(budgetScore),
      transit:   Math.round(transitScore),
      moveIn:    Math.round(moveInScore),
      amenities: Math.round(amenityScore),
    },
    conflicts,
    sharedNeighborhoods,
  };
}
```

## Part 4 — Conflict Detection Before Group Forms

Before a renter can invite someone to a group, or before an agent groups renters together, run a conflict check:

```ts
export function detectGroupConflicts(
  members: Array<User & { apartmentPrefs: ApartmentPreferences }>
): {
  hasConflicts: boolean;
  conflicts: string[];
  suggestions: string[];
  sharedNeighborhoods: string[];
} {
  const conflicts: string[] = [];
  const suggestions: string[] = [];

  // Check all pairs
  const pairs = getPairs(members);
  let allSharedNeighborhoods: string[] = [];

  for (const [a, b] of pairs) {
    const result = calculateGroupCompatibilityScore(a, b);
    conflicts.push(...result.conflicts);

    if (allSharedNeighborhoods.length === 0) {
      allSharedNeighborhoods = result.sharedNeighborhoods;
    } else {
      // Intersection — only keep neighborhoods that work for ALL pairs
      allSharedNeighborhoods = allSharedNeighborhoods.filter(n =>
        result.sharedNeighborhoods.includes(n)
      );
    }
  }

  // Check bedroom count vs group size
  const desiredBedrooms = members[0]?.apartmentPrefs?.desiredBedrooms;
  const roomatesNeeded = desiredBedrooms; // 2BR needs 2 people, etc.
  if (members.length > roomatesNeeded) {
    conflicts.push(`Group has ${members.length} people but you're looking for a ${desiredBedrooms}BR`);
    suggestions.push(`Remove ${members.length - roomatesNeeded} member(s) or switch to a ${members.length}BR search`);
  }

  // Budget conflict
  const budgetConflicts = conflicts.filter(c => c.includes('Budget'));
  if (budgetConflicts.length > 0) {
    const budgets = members.map(m =>
      `${m.full_name}: $${m.apartmentPrefs.budgetPerPersonMin}–$${m.apartmentPrefs.budgetPerPersonMax}`
    ).join(', ');
    suggestions.push(`Budgets don't fully overlap: ${budgets}. Consider finding someone in the same range.`);
  }

  // Transit conflict with compromise suggestion
  if (allSharedNeighborhoods.length === 0 && !conflicts.some(c => c.includes('flexible'))) {
    const allTrains = [...new Set(members.flatMap(m => m.apartmentPrefs.preferredTrains))];
    // Try dropping one member's strictest requirement to find compromise
    for (const member of members) {
      const otherTrains = [...new Set(
        members
          .filter(m => m.id !== member.id)
          .flatMap(m => m.apartmentPrefs.preferredTrains)
      )];
      const compromise = findCompatibleNeighborhoods(otherTrains);
      if (compromise.length > 0) {
        suggestions.push(
          `If ${member.full_name} is flexible on their commute, ${compromise.join(', ')} could work for everyone else`
        );
        break;
      }
    }
  }

  return {
    hasConflicts: conflicts.length > 0,
    conflicts: [...new Set(conflicts)],
    suggestions,
    sharedNeighborhoods: allSharedNeighborhoods,
  };
}
```

## Part 5 — AI Apartment Suggestions for Groups

Once a group is formed and passes conflict detection, the AI suggests apartments.

### Screen: `screens/renter/GroupApartmentSuggestionsScreen.tsx`

```ts
const generateApartmentSuggestions = async (
  group: Group,
  members: Array<User & { apartmentPrefs: ApartmentPreferences }>
) => {
  // Compile group requirements
  const desiredBedrooms = members[0].apartmentPrefs.desiredBedrooms;
  const allTrains = [...new Set(members.flatMap(m => m.apartmentPrefs.preferredTrains))];
  const compatibleNeighborhoods = findCompatibleNeighborhoods(allTrains);

  // Combined budget = sum of each person's max
  const combinedBudgetMin = members.reduce((sum, m) => sum + m.apartmentPrefs.budgetPerPersonMin, 0);
  const combinedBudgetMax = members.reduce((sum, m) => sum + m.apartmentPrefs.budgetPerPersonMax, 0);

  // Must-have amenities that ALL members share
  const sharedAmenities = members[0].apartmentPrefs.amenityMustHaves.filter(a =>
    members.every(m => m.apartmentPrefs.amenityMustHaves.includes(a))
  );

  // Earliest move-in that works for everyone
  const latestMoveIn = members
    .map(m => new Date(m.apartmentPrefs.moveInDate))
    .sort((a, b) => b.getTime() - a.getTime())[0];

  // Fetch listings that match
  const { data: listings } = await supabase
    .from('listings')
    .select('*')
    .eq('bedrooms', desiredBedrooms)
    .gte('price', combinedBudgetMin)
    .lte('price', combinedBudgetMax)
    .in('neighborhood', compatibleNeighborhoods)
    .lte('available_date', latestMoveIn.toISOString())
    .eq('status', 'active')
    .order('is_featured', { ascending: false })
    .limit(10);

  // Score each listing
  return (listings ?? []).map(listing => {
    const trainScore = scoreListing_Transit(listing, allTrains);
    const amenityScore = scoreListing_Amenities(listing, sharedAmenities);
    const budgetScore = scoreListing_Budget(listing, combinedBudgetMin, combinedBudgetMax, members.length);

    const totalScore = Math.round(
      trainScore  * 0.40 +
      budgetScore * 0.35 +
      amenityScore * 0.25
    );

    const perPersonRent = Math.round(listing.price / members.length);
    const trainsNearby = NEIGHBORHOOD_TRAINS[listing.neighborhood] ?? [];

    return {
      listing,
      totalScore,
      perPersonRent,
      trainsNearby,
      breakdown: { trainScore, amenityScore, budgetScore },
      aiReason: generateListingReason(listing, members, trainScore, budgetScore, perPersonRent),
    };
  }).sort((a, b) => b.totalScore - a.totalScore).slice(0, 5);
};

function generateListingReason(
  listing: any,
  members: any[],
  trainScore: number,
  budgetScore: number,
  perPersonRent: number
): string {
  const trains = NEIGHBORHOOD_TRAINS[listing.neighborhood] ?? [];
  const memberNames = members.map(m => m.full_name.split(' ')[0]).join(' and ');

  if (trainScore >= 90 && budgetScore >= 90) {
    return `Perfect commute and budget fit for ${memberNames}. ${trains.join('/')} trains right in the neighborhood.`;
  }
  if (trainScore >= 90) {
    return `Great transit access — ${trains.join('/')} trains nearby. Budget is tight but workable at $${perPersonRent}/person.`;
  }
  if (budgetScore >= 90) {
    return `Strong value at $${perPersonRent}/person. ${trains[0]} train is a short walk — worth checking the commute time.`;
  }
  return `Solid option in ${listing.neighborhood}. Check if the commute works for everyone before scheduling a showing.`;
}

function scoreListing_Transit(listing: any, requiredTrains: string[]): number {
  const nearbyTrains = NEIGHBORHOOD_TRAINS[listing.neighborhood] ?? [];
  const directMatches = requiredTrains.filter(t => nearbyTrains.includes(t));
  return Math.min(100, (directMatches.length / requiredTrains.length) * 100);
}

function scoreListing_Budget(
  listing: any,
  combinedMin: number,
  combinedMax: number,
  memberCount: number
): number {
  if (listing.price < combinedMin) return 60; // under budget — might mean something's wrong
  if (listing.price > combinedMax) return 0;  // over budget — excluded
  const position = (combinedMax - listing.price) / (combinedMax - combinedMin);
  return Math.round(60 + position * 40); // 60–100 range
}

function scoreListing_Amenities(listing: any, mustHaves: string[]): number {
  if (mustHaves.length === 0) return 70;
  const matched = mustHaves.filter(a => listing.amenities?.includes(a));
  return Math.round((matched.length / mustHaves.length) * 100);
}
```

## Part 6 — Updated Agent Matchmaker with Transit Pre-Filtering

### Key change:
When an agent selects a listing to find renters for, the browse feed is **pre-filtered to only show renters whose preferred train lines are served by the listing's neighborhood.** The agent never sees renters who couldn't get to work from that apartment.

```ts
// In BrowseRentersScreen — when agent selects a listing
const loadRentersForListing = async (listing: Listing) => {
  // Get trains that serve this listing's neighborhood
  const listingTrains = NEIGHBORHOOD_TRAINS[listing.neighborhood] ?? [];

  // Fetch ALL renters in that city
  const { data: allRenters } = await supabase
    .from('users')
    .select(`
      id, full_name, avatar_url, occupation, age, city, neighborhood,
      profiles (
        budget_per_person_min, budget_per_person_max,
        move_in_date, desired_bedrooms,
        preferred_trains, preferred_neighborhoods,
        amenity_must_haves, location_flexible, wfh,
        cleanliness, noise_tolerance, sleep_schedule,
        smoking, drinking, pets, interests
      )
    `)
    .eq('role', 'renter')
    .eq('city', listing.city)
    .eq('profiles.apartment_prefs_complete', true)
    .not('id', 'in', `(${agentShortlistedIds.join(',')})`);

  // Pre-filter: transit must work
  const transitCompatible = allRenters?.filter(renter => {
    const prefs = renter.profiles;
    if (!prefs) return false;
    if (prefs.location_flexible || prefs.wfh) return true;

    // At least one of renter's trains must serve this neighborhood
    return prefs.preferred_trains?.some((train: string) =>
      listingTrains.includes(train) || hasReasonableTransfer(train, listingTrains)
    );
  }) ?? [];

  // Pre-filter: budget must cover share
  const sharePerPerson = listing.rent / listing.bedrooms;
  const budgetCompatible = transitCompatible.filter(renter => {
    const maxBudget = renter.profiles?.budget_per_person_max ?? 0;
    return maxBudget >= sharePerPerson;
  });

  // Pre-filter: bedroom match
  const bedroomCompatible = budgetCompatible.filter(renter => {
    const desired = renter.profiles?.desired_bedrooms;
    if (!desired) return true;
    return desired === listing.bedrooms;
  });

  // Pre-filter: move-in date within 30 days of listing availability
  const dateCompatible = bedroomCompatible.filter(renter => {
    if (!renter.profiles?.move_in_date || !listing.available_date) return true;
    const moveIn = new Date(renter.profiles.move_in_date);
    const available = new Date(listing.available_date);
    const diffDays = Math.abs((moveIn.getTime() - available.getTime()) / (1000 * 60 * 60 * 24));
    return diffDays <= 30;
  });

  setFilteredRenters(dateCompatible);

  // Show agent what was filtered
  setFilterSummary({
    total: allRenters?.length ?? 0,
    afterTransit: transitCompatible.length,
    afterBudget: budgetCompatible.length,
    afterBedrooms: bedroomCompatible.length,
    afterDate: dateCompatible.length,
  });
};
```

### AI group suggestion for agent (transit-aware):
```ts
const generateAgentGroupSuggestion = (
  shortlisted: Renter[],
  listing: Listing
) => {
  const needed = listing.bedrooms;
  const combinations = getCombinations(shortlisted, needed);

  const scored = combinations.map(group => {
    // All transit already compatible (pre-filtered)
    // Check compatibility between group members
    const pairs = getPairs(group);
    const avgCompatibility = pairs.length > 0
      ? pairs.reduce((sum, [a, b]) => {
          const result = calculateGroupCompatibilityScore(a, b);
          return sum + result.total;
        }, 0) / pairs.length
      : 0;

    // Combined budget
    const combinedMax = group.reduce((sum, r) =>
      sum + (r.profiles?.budget_per_person_max ?? 0), 0);
    const coversRent = combinedMax >= listing.rent;

    // Move-in alignment
    const moveInDates = group.map(r => new Date(r.profiles?.move_in_date ?? Date.now()));
    const maxDiff = Math.max(...moveInDates.map(d =>
      Math.abs(d.getTime() - new Date(listing.available_date).getTime())
    )) / (1000 * 60 * 60 * 24);
    const moveInAligned = maxDiff <= 14;

    return {
      group,
      avgCompatibility: Math.round(avgCompatibility),
      combinedBudgetMax: combinedMax,
      coversRent,
      moveInAligned,
      score: (avgCompatibility * 0.6) +
             (coversRent ? 30 : 0) +
             (moveInAligned ? 10 : 0),
    };
  });

  return scored.sort((a, b) => b.score - a.score)[0] ?? null;
};
```

## Complete Feature Flow Summary

### Renter flow:
```
1. Complete apartment preferences questionnaire
   (bedrooms, budget per person, train lines, move-in, amenities)
         ↓
2. App shows compatible group matches
   (people who want same bedroom count, compatible trains,
   overlapping budget, similar move-in)
         ↓
3. AI checks for conflicts before invite is sent
   (flags budget mismatch, transit incompatibility, date gaps)
         ↓
4. Group forms — already aligned on everything
         ↓
5. AI suggests top 5 apartments
   (filtered by all members' trains, combined budget, shared amenities)
         ↓
6. Group votes on each suggestion
         ↓
7. Group schedules showings on agreed apartments
```

### Agent flow:
```
1. Agent selects a listing to fill
         ↓
2. Browse feed pre-filters to only show renters who:
   - Have a train line that serves the listing's neighborhood
   - Can afford their share of the rent
   - Want that bedroom count
   - Are available by the listing's date
         ↓
3. AI surfaces top picks from pre-filtered renters
   (ranked by profile quality + lifestyle compatibility)
         ↓
4. Agent shortlists compatible renters
         ↓
5. AI generates best group combination
   (highest avg compatibility + covers rent + aligned move-in)
         ↓
6. Agent sends group invites
         ↓
7. Renters accept/decline
         ↓
8. Group is placed → placement fee charged
```
