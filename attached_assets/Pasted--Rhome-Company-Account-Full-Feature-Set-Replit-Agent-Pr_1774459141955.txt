## Overview

This prompt implements all company account features:

1. **Sign-up form** — Full Name + Company Name fields for company accounts
2. **Multi-seat team access** — Owner can invite team members with their own logins
3. **Team roles** — Owner, Admin, Member with different permission levels
4. **Team management dashboard** — Invite, remove, and manage team members
5. **Seat limits by plan** — Starter: 3 seats, Pro: 10 seats, Enterprise: unlimited

## Part 1 — Database Migration

Create `033_company_teams.sql` and run it in the Supabase SQL Editor:

```sql
-- Add company_name and full_name to users table
ALTER TABLE public.users
  ADD COLUMN IF NOT EXISTS company_name TEXT,
  ADD COLUMN IF NOT EXISTS full_name_owner TEXT;

-- Team members table
CREATE TABLE IF NOT EXISTS public.team_members (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  company_user_id UUID NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,
  member_user_id UUID REFERENCES public.users(id) ON DELETE SET NULL,
  email TEXT NOT NULL,
  full_name TEXT,
  role TEXT NOT NULL DEFAULT 'member' CHECK (role IN ('owner', 'admin', 'member')),
  status TEXT NOT NULL DEFAULT 'pending' CHECK (status IN ('pending', 'active', 'removed')),
  invited_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  joined_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Index for fast lookups
CREATE INDEX IF NOT EXISTS idx_team_members_company ON team_members(company_user_id);
CREATE INDEX IF NOT EXISTS idx_team_members_email ON team_members(email);

-- RLS
ALTER TABLE public.team_members ENABLE ROW LEVEL SECURITY;

-- Company owner can see all their team members
CREATE POLICY "owner_see_team" ON public.team_members
  FOR SELECT USING (company_user_id = auth.uid());

-- Team member can see their own record
CREATE POLICY "member_see_own" ON public.team_members
  FOR SELECT USING (member_user_id = auth.uid());

-- Only owner/admin can insert (invite)
CREATE POLICY "owner_invite" ON public.team_members
  FOR INSERT WITH CHECK (company_user_id = auth.uid());

-- Only owner/admin can update
CREATE POLICY "owner_update" ON public.team_members
  FOR UPDATE USING (company_user_id = auth.uid());
```

## Part 4 — Team Members: AuthContext Functions

Add these functions to `AuthContext.tsx`:

```typescript
// Get all team members for the current company account
const getTeamMembers = async (): Promise<TeamMember[]> => {
  if (!user || user.hostType !== 'company') return [];
  const { data, error } = await supabase
    .from('team_members')
    .select('*')
    .eq('company_user_id', user.id)
    .neq('status', 'removed')
    .order('created_at', { ascending: true });
  if (error) return [];
  return data as TeamMember[];
};

// Invite a new team member by email
const inviteTeamMember = async (email: string, name: string, role: 'admin' | 'member'): Promise<void> => {
  if (!user || user.hostType !== 'company') return;

  // Check seat limit
  const members = await getTeamMembers();
  const seatLimit = getTeamSeatLimit(user.hostSubscription?.plan ?? 'free');
  if (members.length >= seatLimit) {
    throw new Error(`Your plan allows up to ${seatLimit} team members. Upgrade to add more.`);
  }

  await supabase.from('team_members').insert({
    company_user_id: user.id,
    email,
    full_name: name,
    role,
    status: 'pending',
  });

  // TODO: Send invite email via Supabase Edge Function
};

// Remove a team member
const removeTeamMember = async (memberId: string): Promise<void> => {
  if (!user) return;
  await supabase
    .from('team_members')
    .update({ status: 'removed' })
    .eq('id', memberId)
    .eq('company_user_id', user.id);
};

// Change a team member's role
const updateTeamMemberRole = async (memberId: string, role: 'admin' | 'member'): Promise<void> => {
  if (!user) return;
  await supabase
    .from('team_members')
    .update({ role })
    .eq('id', memberId)
    .eq('company_user_id', user.id);
};

// Get seat limit based on plan
function getTeamSeatLimit(plan: string): number {
  switch (plan) {
    case 'company_starter': return 3;
    case 'company_pro': return 10;
    case 'company_enterprise': return Infinity;
    default: return 1; // free plan — owner only
  }
}
```

Add a `TeamMember` type:
```typescript
interface TeamMember {
  id: string;
  companyUserId: string;
  memberUserId?: string;
  email: string;
  fullName?: string;
  role: 'owner' | 'admin' | 'member';
  status: 'pending' | 'active' | 'removed';
  invitedAt: string;
  joinedAt?: string;
}
```

Export all four functions from `AuthContext`.

## Part 8 — Company Name on Listings

Everywhere a host's name is displayed on a listing or profile, show the company name instead of the personal name for company accounts:

```typescript
function getDisplayName(user: User): string {
  if (user.hostType === 'company' && user.companyName) {
    return user.companyName;
  }
  return user.name;
}
```

- Listing cards
