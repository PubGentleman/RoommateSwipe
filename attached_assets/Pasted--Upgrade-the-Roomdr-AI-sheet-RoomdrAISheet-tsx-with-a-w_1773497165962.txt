function getInsightBorderColor(insightType: string, value?: number): string {
  // RED/CORAL — warnings and declining metrics
  if (insightType === 'match_pool_impact') return '#FF6B6B';
  if (insightType === 'match_rate' && value !== undefined && value < 0) 
    return '#FF6B6B'; // declining match rate
  if (insightType === 'response_rate' && value !== undefined && value < 70) 
    return '#FF6B6B';
  if (insightType === 'move_in_timeline') return '#FF6B6B';
  if (insightType === 'safety_briefing') return '#FF6B6B';

  // GREEN — positive/healthy metrics
  if (insightType === 'profile_completion' && value !== undefined && value >= 90) 
    return '#4CAF50';
  if (insightType === 'match_rate' && value !== undefined && value > 0) 
    return '#4CAF50';
  if (insightType === 'response_rate' && value !== undefined && value >= 70) 
    return '#4CAF50';

  // GREY — neutral/informational
  return '#3A3A3A';
}
