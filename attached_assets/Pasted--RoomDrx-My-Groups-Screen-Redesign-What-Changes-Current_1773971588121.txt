# RoomDrx — My Groups Screen Redesign

## What Changes

| Current | Redesigned |
|---------|-----------|
| Plain underline tabs | Pill-style sliding tab bar |
| Flat invite code input | Frosted glass invite card |
| Tall text-heavy group cards | Compact visual cards with gradient accent |
| Full member list on card | Overlapping avatar stack + slot progress bar |
| No quick actions | Chat and Info buttons directly on card |
| Raw ✕ remove buttons on main list | Removed — belongs in GroupInfo only |

import React, { useState, useEffect, useRef } from 'react';
import {
  View, ScrollView, TextInput, TouchableOpacity,
  FlatList, ActivityIndicator, StyleSheet, Animated,
  Dimensions, Alert,
} from 'react-native';
import { LinearGradient } from 'expo-linear-gradient';
import { Feather } from '@expo/vector-icons';
import { ThemedText } from '../../components/ThemedText';
import { useTheme } from '../../context/ThemeContext';
import { Typography, Spacing } from '../../constants/theme';
import { useAuth } from '../../context/AuthContext';
import { getMyGroups, joinGroupByCode } from '../../services/groupService';

const { width: SCREEN_WIDTH } = Dimensions.get('window');
const TABS = ['My Groups', 'Discover', 'Create'];

export default function GroupsScreen({ navigation }: any) {
  const { theme } = useTheme();
  const { user } = useAuth();

  const [activeTab, setActiveTab] = useState(0);
  const [groups, setGroups] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);
  const [inviteCode, setInviteCode] = useState('');
  const [joining, setJoining] = useState(false);

  // Animated pill for tab indicator
  const pillAnim = useRef(new Animated.Value(0)).current;

  useEffect(() => {
    loadGroups();
  }, []);

  async function loadGroups() {
    try {
      const data = await getMyGroups(user!.id);
      setGroups(data);
    } finally {
      setLoading(false);
    }
  }

  function handleTabPress(index: number) {
    setActiveTab(index);
    Animated.spring(pillAnim, {
      toValue: index,
      useNativeDriver: true,
      tension: 80,
      friction: 10,
    }).start();

    if (index === 2) {
      navigation.navigate('CreateGroup');
      // Reset tab back after navigation
      setTimeout(() => {
        setActiveTab(0);
        Animated.spring(pillAnim, { toValue: 0, useNativeDriver: true }).start();
      }, 300);
    }
  }

  async function handleJoin() {
    const code = inviteCode.trim().toUpperCase();
    if (code.length !== 6) {
      Alert.alert('Invalid Code', 'Invite codes are 6 characters long.');
      return;
    }
    setJoining(true);
    try {
      await joinGroupByCode(code);
      setInviteCode('');
      loadGroups();
    } catch (e: any) {
      const msg = e.message || '';
      if (msg === 'GROUP_FULL') Alert.alert('Group Full', 'This group has reached its member limit.');
      else if (msg === 'ALREADY_MEMBER') Alert.alert('Already a Member', 'You are already in this group.');
      else if (msg === 'NOT_FOUND') Alert.alert('Code Not Found', 'This invite code does not exist.');
      else Alert.alert('Error', 'Could not join group. Please try again.');
    } finally {
      setJoining(false);
    }
  }

  // Tab pill translateX
  const TAB_WIDTH = (SCREEN_WIDTH - Spacing.md * 2) / 3;
  const pillTranslateX = pillAnim.interpolate({
    inputRange: [0, 1, 2],
    outputRange: [0, TAB_WIDTH, TAB_WIDTH * 2],
  });

  return (
    <View style={{ flex: 1, backgroundColor: theme.background }}>
      {/* ── PAGE HEADER ──────────────────────────────────────────── */}
      <View style={styles.pageHeader}>
        <ThemedText style={Typography.h2}>Groups</ThemedText>
        <TouchableOpacity
          style={[styles.notifBtn, { backgroundColor: theme.card, borderColor: theme.border }]}
          onPress={() => navigation.navigate('Notifications')}
        >
          <Feather name="bell" size={18} color={theme.text} />
        </TouchableOpacity>
      </View>

      {/* ── TAB BAR ──────────────────────────────────────────────── */}
      <View style={[styles.tabBar, { backgroundColor: theme.card, borderColor: theme.border }]}>
        {/* Animated pill */}
        <Animated.View
          style={[
            styles.tabPill,
            { backgroundColor: theme.primary, width: TAB_WIDTH - 8, transform: [{ translateX: pillTranslateX }] },
          ]}
        />
        {TABS.map((tab, i) => (
          <TouchableOpacity
            key={tab}
            style={[styles.tabBtn, { width: TAB_WIDTH }]}
            onPress={() => handleTabPress(i)}
          >
            <ThemedText
              style={[
                styles.tabLabel,
                { color: activeTab === i ? '#fff' : theme.textSecondary },
              ]}
            >
              {tab}
            </ThemedText>
          </TouchableOpacity>
        ))}
      </View>

      {/* ── CONTENT ──────────────────────────────────────────────── */}
      {activeTab === 0 && (
        <ScrollView
          contentContainerStyle={{ padding: Spacing.md, paddingBottom: 100 }}
          showsVerticalScrollIndicator={false}
        >
          {/* ── INVITE CODE CARD ─────────────────────────────────── */}
          <View style={[styles.inviteCard, { backgroundColor: theme.card, borderColor: theme.border }]}>
            <View style={[styles.inviteIconCircle, { backgroundColor: theme.primary + '20' }]}>
              <Feather name="hash" size={16} color={theme.primary} />
            </View>
            <TextInput
              value={inviteCode}
              onChangeText={v => setInviteCode(v.toUpperCase())}
              placeholder="Enter invite code"
              placeholderTextColor={theme.textSecondary}
              autoCapitalize="characters"
              maxLength={6}
              style={[styles.inviteInput, { color: theme.text }]}
            />
            <TouchableOpacity
              style={[styles.joinBtn, { backgroundColor: inviteCode.length === 6 ? theme.primary : theme.border }]}
              onPress={handleJoin}
              disabled={joining || inviteCode.length !== 6}
            >
              {joining
                ? <ActivityIndicator size="small" color="#fff" />
                : <ThemedText style={styles.joinBtnText}>Join</ThemedText>
              }
            </TouchableOpacity>
          </View>

          {/* ── SECTION LABEL ────────────────────────────────────── */}
          <View style={styles.sectionRow}>
            <ThemedText style={[styles.sectionLabel, { color: theme.textSecondary }]}>
              MY ROOMMATE GROUPS
            </ThemedText>
            <ThemedText style={[Typography.small, { color: theme.textSecondary }]}>
              {groups.length} {groups.length === 1 ? 'group' : 'groups'}
            </ThemedText>
          </View>

          {/* ── GROUP CARDS ──────────────────────────────────────── */}
          {loading ? (
            <ActivityIndicator color={theme.primary} style={{ marginTop: 40 }} />
          ) : groups.length === 0 ? (
            <EmptyState theme={theme} onCreatePress={() => navigation.navigate('CreateGroup')} />
          ) : (
            groups.map(group => (
              <GroupCard
                key={group.id}
                group={group}
                currentUserId={user!.id}
                theme={theme}
                onChatPress={() => navigation.navigate('GroupChat', {
                  groupId: group.id,
                  groupName: group.name,
                })}
                onInfoPress={() => navigation.navigate('GroupInfo', { groupId: group.id })}
              />
            ))
          )}
        </ScrollView>
      )}

      {activeTab === 1 && (
        <DiscoverTab theme={theme} navigation={navigation} />
      )}
    </View>
  );
}

// ─── Group Card ──────────────────────────────────────────────────────────────

function GroupCard({
  group, currentUserId, theme, onChatPress, onInfoPress,
}: {
  group: any;
  currentUserId: string;
  theme: any;
  onChatPress: () => void;
  onInfoPress: () => void;
}) {
  const filled = group.memberCount;
  const total = group.memberLimit ?? 4;
  const fillPercent = filled / total;
  const isAdmin = group.adminId === currentUserId;
  const hasUnread = group.unreadCount > 0;

  return (
    <TouchableOpacity
      style={[styles.groupCard, { backgroundColor: theme.card, borderColor: theme.border }]}
      onPress={onChatPress}
      activeOpacity={0.85}
    >
      {/* Coral left accent bar */}
      <View style={[styles.accentBar, { backgroundColor: theme.primary }]} />

      <View style={{ flex: 1, paddingLeft: 14 }}>

        {/* ── TOP ROW: avatar + name + badges ─────────────────── */}
        <View style={styles.cardTopRow}>
          {/* Group avatar — overlapping member photos */}
          <View style={styles.avatarStack}>
            {group.members.slice(0, 3).map((m: any, i: number) => (
              <View
                key={m.id}
                style={[
                  styles.stackAvatar,
                  {
                    backgroundColor: theme.primary,
                    borderColor: theme.card,
                    zIndex: 3 - i,
                    marginLeft: i === 0 ? 0 : -10,
                  },
                ]}
              >
                <ThemedText style={{ fontSize: 11, fontWeight: '800', color: '#fff' }}>
                  {m.name.charAt(0).toUpperCase()}
                </ThemedText>
              </View>
            ))}
            {group.memberCount > 3 && (
              <View style={[styles.stackAvatar, styles.stackOverflow, { backgroundColor: theme.border, borderColor: theme.card, marginLeft: -10 }]}>
                <ThemedText style={{ fontSize: 9, fontWeight: '700', color: theme.textSecondary }}>
                  +{group.memberCount - 3}
                </ThemedText>
              </View>
            )}
          </View>

          {/* Name + member count */}
          <View style={{ flex: 1, marginLeft: 10 }}>
            <View style={styles.nameRow}>
              <ThemedText style={[Typography.body, { fontWeight: '800' }]} numberOfLines={1}>
                {group.name}
              </ThemedText>
              {isAdmin && (
                <View style={[styles.adminPill, { backgroundColor: theme.primary + '25' }]}>
                  <ThemedText style={{ fontSize: 10, color: theme.primary, fontWeight: '700' }}>
                    Admin
                  </ThemedText>
                </View>
              )}
            </View>

            {/* Member slots progress */}
            <View style={styles.progressRow}>
              <View style={[styles.progressTrack, { backgroundColor: theme.border }]}>
                <View
                  style={[
                    styles.progressFill,
                    {
                      backgroundColor: fillPercent >= 1 ? '#22C55E' : theme.primary,
                      width: `${Math.min(fillPercent * 100, 100)}%`,
                    },
                  ]}
                />
              </View>
              <ThemedText style={[styles.slotText, { color: theme.textSecondary }]}>
                {filled}/{total}
              </ThemedText>
            </View>
          </View>

          {/* Unread badge */}
          {hasUnread && (
            <View style={[styles.unreadBadge, { backgroundColor: theme.primary }]}>
              <ThemedText style={styles.unreadText}>
                {group.unreadCount > 9 ? '9+' : group.unreadCount}
              </ThemedText>
            </View>
          )}
        </View>

        {/* ── DESCRIPTION ─────────────────────────────────────── */}
        <ThemedText
          style={[Typography.small, { color: theme.textSecondary, marginTop: 8, marginBottom: 8 }]}
          numberOfLines={2}
        >
          {group.description || 'No description yet.'}
        </ThemedText>

        {/* ── META CHIPS ──────────────────────────────────────── */}
        <View style={styles.chipRow}>
          {group.minBudget && (
            <View style={[styles.chip, { backgroundColor: theme.primary + '15', borderColor: theme.primary + '30' }]}>
              <Feather name="dollar-sign" size={10} color={theme.primary} />
              <ThemedText style={[styles.chipText, { color: theme.primary }]}>
                ${group.minBudget.toLocaleString()}/mo
              </ThemedText>
            </View>
          )}
          {group.targetNeighborhood && (
            <View style={[styles.chip, { backgroundColor: theme.border + '60', borderColor: theme.border }]}>
              <Feather name="map-pin" size={10} color={theme.textSecondary} />
              <ThemedText style={[styles.chipText, { color: theme.textSecondary }]}>
                {group.targetNeighborhood}
              </ThemedText>
            </View>
          )}
          {group.linkedListing && (
            <View style={[styles.chip, { backgroundColor: '#22C55E15', borderColor: '#22C55E40' }]}>
              <Feather name="home" size={10} color="#22C55E" />
              <ThemedText style={[styles.chipText, { color: '#22C55E' }]}>
                {group.linkedListing.status === 'rented' ? 'Rented' : 'Property Linked'}
              </ThemedText>
            </View>
          )}
        </View>

        {/* ── BOTTOM ACTION ROW ───────────────────────────────── */}
        <View style={[styles.actionRow, { borderTopColor: theme.border }]}>
          {/* Last message preview */}
          {group.lastMessage ? (
            <ThemedText
              style={[Typography.small, { color: theme.textSecondary, flex: 1 }]}
              numberOfLines={1}
            >
              {group.lastMessage.senderName}: {group.lastMessage.content}
            </ThemedText>
          ) : (
            <ThemedText style={[Typography.small, { color: theme.textSecondary, flex: 1 }]}>
              No messages yet
            </ThemedText>
          )}

          {/* Quick actions */}
          <View style={{ flexDirection: 'row', gap: 8 }}>
            <TouchableOpacity
              style={[styles.actionBtn, { backgroundColor: theme.primary }]}
              onPress={onChatPress}
            >
              <Feather name="message-circle" size={13} color="#fff" />
              <ThemedText style={styles.actionBtnText}>Chat</ThemedText>
            </TouchableOpacity>
            <TouchableOpacity
              style={[styles.actionBtn, { backgroundColor: theme.border }]}
              onPress={onInfoPress}
            >
              <Feather name="info" size={13} color={theme.text} />
            </TouchableOpacity>
          </View>
        </View>
      </View>
    </TouchableOpacity>
  );
}

## Key Design Changes Explained

### Tab Bar
- Replaced plain underline tabs with a **pill-style tab bar** in a rounded container
- An animated coral pill slides under the active tab using `Animated.spring`
- Tapping "Create" immediately navigates to `CreateGroupScreen` and resets back

### Invite Code Card
- Replaced the bare text input with a **frosted card** containing an icon, spaced input, and a Join button that activates only when 6 characters are entered
- Invite code auto-uppercases as the user types

### Group Cards
- **Coral left accent bar** gives each card a visual anchor
- **Overlapping avatar stack** replaces the verbose member list — shows up to 3 member initials + a "+N" overflow chip
- **Member slot progress bar** shows fill percentage visually (turns green when full)
- **Admin pill badge** next to group name
- **Unread message badge** in the top-right corner
- **Last message preview** at the bottom of the card (like iMessage list)
- **Chat** (coral) and **Info** (neutral) quick-action buttons eliminate the need to navigate just to get to the chat
- **"Property Linked"** green chip appears when a listing is attached to the group

### Empty State
- Friendly icon + message + "Create a Group" CTA button instead of just a blank list
