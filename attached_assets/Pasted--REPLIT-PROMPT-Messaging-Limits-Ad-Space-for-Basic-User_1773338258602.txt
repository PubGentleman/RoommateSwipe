**PART 2 — MESSAGING LIMITS BY TIER**

```ts
export const MESSAGING_LIMITS = {
  basic: {
    dailyMessages: 20,
    activeChats: 3,
    coldMessaging: false,
  },
  plus: {
    dailyMessages: 200,
    activeChats: 10,
    coldMessaging: false,
  },
  elite: {
    dailyMessages: Infinity,
    activeChats: Infinity,
    coldMessaging: true,
  },
```

**PART 3 — MESSAGING LIMIT LOGIC**

Create `utils/messagingUtils.ts`:

```ts
export const canSendMessage = (
  user: User,
  plan: 'basic' | 'plus' | 'elite'
): { allowed: boolean; reason?: string } => {
  const limits = MESSAGING_LIMITS[plan];

  // Reset daily count if new day
  const today = new Date().toISOString().split('T')[0];
  const resetDate = user.messagingData?.dailyMessageResetDate?.split('T')[0];
  const dailyCount = resetDate === today 
    ? (user.messagingData?.dailyMessageCount ?? 0) 
    : 0;

  if (plan !== 'elite' && dailyCount >= limits.dailyMessages) {
    return {
      allowed: false,
      reason: `daily_limit`
    };
  }

  return { allowed: true };
};

export const canStartNewChat = (
  user: User,
  plan: 'basic' | 'plus' | 'elite'
): { allowed: boolean; reason?: string } => {
  const limits = MESSAGING_LIMITS[plan];
  const activeChats = user.messagingData?.activeChatsCount ?? 0;

  if (activeChats >= limits.activeChats) {
    return {
      allowed: false,
      reason: 'chat_limit'
    };
  }

  return { allowed: true };
};
```

**PART 9 — RESET LOGIC**

On app launch and whenever the app comes to foreground, run this check in `AuthContext`:

```ts
const today = new Date().toISOString().split('T')[0];
const resetDate = user.messagingData?.dailyMessageResetDate?.split('T')[0];

if (resetDate !== today) {
  await supabase.from('users').update({
    'messagingData.dailyMessageCount': 0,
    'messagingData.dailyMessageResetDate': new Date().toISOString()
  }).eq('id', user.id);
}
```
