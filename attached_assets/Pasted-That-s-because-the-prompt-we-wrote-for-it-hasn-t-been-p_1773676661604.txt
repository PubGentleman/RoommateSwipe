```ts
static async getVisibleRenterGroups(): Promise<RenterGroupCard[]> {
  try {
    // Try Supabase
    const { data, error } = await supabase
      .from('groups')
      .select('*')
      .eq('type', 'roommate')
      .eq('visible_to_hosts', true)
      .order('created_at', { ascending: false });
    if (error || !data?.length) throw new Error('No data');
    return data.map(g => ({
      groupId: g.id,
      memberCount: g.member_count ?? 2,
      budgetMin: g.budget_min ?? 1500,
      budgetMax: g.budget_max ?? 2500,
      moveInDate: g.move_in_date ?? 'Flexible',
      neighborhoods: g.neighborhoods ?? [],
      lifestyleTags: g.lifestyle_tags ?? [],
      occupationTypes: g.occupation_types ?? [],
      createdAt: g.created_at,
    }));
  } catch {
    // Return empty so screen uses its own isDev fallback
    return [];
  }
}
```
