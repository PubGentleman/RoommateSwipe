# RoomDrx — Host Type Differentiation (Individual / Company / Agent)

## Overview

All hosts share a single `host` role. A `host_type` field on the profile determines how their account looks, what features they see, and whether match percentage is shown. Renters see a different profile card and contact UI depending on who listed the property.

---

## Database Changes

### Add host type fields to `profiles` table

```sql
ALTER TABLE profiles
  ADD COLUMN IF NOT EXISTS host_type TEXT DEFAULT 'individual'
    CHECK (host_type IN ('individual', 'company', 'agent')),
  ADD COLUMN IF NOT EXISTS company_name TEXT,
  ADD COLUMN IF NOT EXISTS company_logo_url TEXT,
  ADD COLUMN IF NOT EXISTS license_number TEXT,
  ADD COLUMN IF NOT EXISTS agency_name TEXT,
  ADD COLUMN IF NOT EXISTS units_managed INT,
  ADD COLUMN IF NOT EXISTS verified_business BOOLEAN DEFAULT false,
  ADD COLUMN IF NOT EXISTS verified_business_at TIMESTAMPTZ,
  ADD COLUMN IF NOT EXISTS avg_response_hours FLOAT;
```

```tsx
// screens/host/onboarding/HostTypeSelectScreen.tsx

import React, { useState } from 'react';
import {
  View, TouchableOpacity, ActivityIndicator, StyleSheet,
} from 'react-native';
import { Feather } from '@expo/vector-icons';
import { ThemedText } from '../../../components/ThemedText';
import { useTheme } from '../../../context/ThemeContext';
import { Typography, Spacing } from '../../../constants/theme';
import { supabase } from '../../../lib/supabase';
import { useAuth } from '../../../context/AuthContext';

const HOST_TYPES = [
  {
    id: 'individual',
    icon: 'user',
    label: 'Individual',
    description: "I'm renting my own property",
    color: '#6C63FF',
  },
  {
    id: 'company',
    icon: 'briefcase',
    label: 'Company',
    description: 'I manage multiple properties',
    color: '#22C55E',
  },
  {
    id: 'agent',
    icon: 'award',
    label: 'Licensed Agent',
    description: "I'm a real estate agent",
    color: '#F59E0B',
  },
] as const;

export default function HostTypeSelectScreen({ navigation }: any) {
  const { theme } = useTheme();
  const { user } = useAuth();
  const [selected, setSelected] = useState<string | null>(null);
  const [saving, setSaving] = useState(false);

  async function handleContinue() {
    if (!selected) return;
    setSaving(true);
    try {
      await supabase
        .from('profiles')
        .update({ host_type: selected })
        .eq('id', user!.id);

      // Route to appropriate next onboarding step
      if (selected === 'company') {
        navigation.replace('HostCompanySetup');
      } else if (selected === 'agent') {
        navigation.replace('HostAgentSetup');
      } else {
        navigation.replace('HostIndividualSetup');
      }
    } catch (e) {
      console.error(e);
    } finally {
      setSaving(false);
    }
  }

  return (
    <View style={{ flex: 1, backgroundColor: theme.background, padding: Spacing.lg }}>
      <ThemedText style={[Typography.h2, { marginBottom: 6 }]}>
        What describes you?
      </ThemedText>
      <ThemedText style={[Typography.body, { color: theme.textSecondary, marginBottom: Spacing.xl }]}>
        This helps us personalise your host experience.
      </ThemedText>

      {HOST_TYPES.map(type => {
        const isSelected = selected === type.id;
        return (
          <TouchableOpacity
            key={type.id}
            style={[
              styles.typeCard,
              {
                backgroundColor: isSelected ? type.color + '15' : theme.card,
                borderColor: isSelected ? type.color : theme.border,
              },
            ]}
            onPress={() => setSelected(type.id)}
          >
            {/* Icon */}
            <View style={[styles.typeIcon, { backgroundColor: type.color + '20' }]}>
              <Feather name={type.icon as any} size={22} color={type.color} />
            </View>

            {/* Text */}
            <View style={{ flex: 1 }}>
              <ThemedText style={[Typography.body, { fontWeight: '700' }]}>
                {type.label}
              </ThemedText>
              <ThemedText style={[Typography.small, { color: theme.textSecondary }]}>
                {type.description}
              </ThemedText>
            </View>

            {/* Check */}
            {isSelected && (
              <View style={[styles.checkCircle, { backgroundColor: type.color }]}>
                <Feather name="check" size={14} color="#fff" />
              </View>
            )}
          </TouchableOpacity>
        );
      })}

      <TouchableOpacity
        style={[
          styles.continueBtn,
          { backgroundColor: selected ? theme.primary : theme.border, marginTop: Spacing.xl },
        ]}
        onPress={handleContinue}
        disabled={!selected || saving}
      >
        {saving
          ? <ActivityIndicator color="#fff" />
          : <ThemedText style={[Typography.body, { color: '#fff', fontWeight: '700' }]}>
              Continue
            </ThemedText>
        }
      </TouchableOpacity>
    </View>
  );
}
```

## Step 3 — `hostTypeUtils.ts` (shared helper)

```ts
// utils/hostTypeUtils.ts

export type HostType = 'individual' | 'company' | 'agent';

export function shouldShowMatchScore(hostType: HostType): boolean {
  return hostType === 'individual';
}

export function getHostBadgeLabel(hostType: HostType): string {
  switch (hostType) {
    case 'company': return 'Property Management';
    case 'agent':   return 'Licensed Agent';
    default:        return '';
  }
}

export function getHostBadgeColor(hostType: HostType): string {
  switch (hostType) {
    case 'company': return '#22C55E';
    case 'agent':   return '#F59E0B';
    default:        return '';
  }
}

export function getHostBadgeIcon(hostType: HostType): string {
  switch (hostType) {
    case 'company': return 'briefcase';
    case 'agent':   return 'award';
    default:        return 'user';
  }
}

export function getContactLabel(hostType: HostType): string {
  switch (hostType) {
    case 'company': return 'Contact Company';
    case 'agent':   return 'Contact Agent';
    default:        return 'Send Message';
  }
}
```

## Step 5 — Match Score Calculation

### Update `matchService.ts`

```ts
import { shouldShowMatchScore, HostType } from '../utils/hostTypeUtils';

export interface MatchScoreResult {
  score: number;           // 0–100
  showScore: boolean;      // false for company/agent
  breakdown: {
    budget: number;
    location: number;
    moveIn: number;
    lifestyle: number;
    zodiac: number;
    verified: number;
  };
}

export function calculateHostRenterMatch(
  listing: any,
  renterProfile: any,
  hostProfile: any
): MatchScoreResult {
  const hostType = hostProfile.host_type as HostType;

  // Companies and agents don't use match score
  if (!shouldShowMatchScore(hostType)) {
    return {
      score: 0,
      showScore: false,
      breakdown: { budget: 0, location: 0, moveIn: 0, lifestyle: 0, zodiac: 0, verified: 0 },
    };
  }

  // ── Individual host match scoring ──────────────────────────

  // Budget (30%) — does renter's budget meet listing price?
  const budgetScore = renterProfile.budget >= listing.price
    ? 100
    : Math.max(0, 100 - ((listing.price - renterProfile.budget) / listing.price) * 100);

  // Location (20%) — does renter's target neighborhood match?
  const locationScore = renterProfile.targetNeighborhood === listing.neighborhood
    ? 100
    : renterProfile.targetCity === listing.city ? 60 : 0;

  // Move-in date (20%) — how close are the dates?
  const moveInScore = calculateMoveInScore(renterProfile.moveInDate, listing.availableFrom);

  // Lifestyle (15%) — cleanliness, noise, pets, smoking
  const lifestyleScore = calculateLifestyleScore(renterProfile, hostProfile);

  // Zodiac (10%) — elemental compatibility
  const zodiacScore = calculateZodiacScore(renterProfile.zodiacSign, hostProfile.zodiacSign);

  // Verified (5%) — bonus for verified profiles
  const verifiedScore = renterProfile.verified ? 100 : 50;

  // Weighted total
  const score = Math.round(
    budgetScore    * 0.30 +
    locationScore  * 0.20 +
    moveInScore    * 0.20 +
    lifestyleScore * 0.15 +
    zodiacScore    * 0.10 +
    verifiedScore  * 0.05
  );

  return {
    score,
    showScore: true,
    breakdown: {
      budget:    Math.round(budgetScore),
      location:  Math.round(locationScore),
      moveIn:    Math.round(moveInScore),
      lifestyle: Math.round(lifestyleScore),
      zodiac:    Math.round(zodiacScore),
      verified:  Math.round(verifiedScore),
    },
  };
}

// ── Move-in scoring ───────────────────────────────────────────
function calculateMoveInScore(renterDate: string, listingDate: string): number {
  if (!renterDate || !listingDate) return 70; // flexible = decent score
  const diffDays = Math.abs(
    (new Date(renterDate).getTime() - new Date(listingDate).getTime()) / (1000 * 60 * 60 * 24)
  );
  if (diffDays <= 7)  return 100;
  if (diffDays <= 14) return 85;
  if (diffDays <= 30) return 65;
  if (diffDays <= 60) return 40;
  return 10;
}

// ── Lifestyle scoring ─────────────────────────────────────────
function calculateLifestyleScore(renter: any, host: any): number {
  let score = 100;
  if (renter.hasPets && !host.petsAllowed)       score -= 40;
  if (renter.smokes  && !host.smokingAllowed)     score -= 40;
  const cleanDiff = Math.abs((renter.cleanlinessLevel ?? 3) - (host.cleanlinessLevel ?? 3));
  score -= cleanDiff * 10;
  const noiseDiff = Math.abs((renter.noiseLevel ?? 3) - (host.noiseLevel ?? 3));
  score -= noiseDiff * 5;
  return Math.max(0, score);
}
```

```tsx
      {getHostBadgeLabel(listing.host.hostType)}
    </ThemedText>
  </View>
)}

{/* Contact button label changes based on type */}
<TouchableOpacity style={[styles.contactBtn, { backgroundColor: theme.primary }]}>
  <ThemedText style={[Typography.body, { color: '#fff', fontWeight: '700' }]}>
    {getContactLabel(listing.host.hostType)}
  </ThemedText>
</TouchableOpacity>
```
