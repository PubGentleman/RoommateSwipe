# RoomDrx — Group Logic Rules

## Overview
This prompt covers four core group behaviors:

1. **Admin leaving** — must promote someone first; group deletes if last member
2. **Removing members** — admin can remove anyone; removed user gets notified
3. **Linked listing rented** — pinned card gets a "Rented" badge; group stays fully active
4. **Group discovery** — opt-in toggle lets groups appear on the listing page for join requests

```sql
CREATE TABLE IF NOT EXISTS group_join_requests (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  group_id UUID REFERENCES groups(id) ON DELETE CASCADE,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  status TEXT DEFAULT 'pending' CHECK (status IN ('pending', 'approved', 'declined')),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(group_id, user_id)
);

ALTER TABLE group_join_requests ENABLE ROW LEVEL SECURITY;

-- User can see their own requests
CREATE POLICY "Users can view own join requests"
ON group_join_requests FOR SELECT
USING (auth.uid() = user_id);

-- Group admins can see requests for their group
CREATE POLICY "Admins can view join requests"
ON group_join_requests FOR SELECT
USING (
  group_id IN (
    SELECT group_id FROM group_members
    WHERE user_id = auth.uid() AND role = 'admin'
  )
);

-- Any authenticated user can submit a request
CREATE POLICY "Users can request to join"
ON group_join_requests FOR INSERT
WITH CHECK (auth.uid() = user_id);

-- Admins can approve or decline
CREATE POLICY "Admins can respond to requests"
ON group_join_requests FOR UPDATE
USING (
  group_id IN (
    SELECT group_id FROM group_members
    WHERE user_id = auth.uid() AND role = 'admin'
  )
);
```

## Logic 1 — Admin Leaving

### Rules
- Admin taps "Leave Group"
- If they are the **only member** → group deletes immediately
- If there are **other members** → show a "Promote someone first" screen before allowing the leave
- Admin picks one member to promote to admin
- After promoting, they can leave
- The promoted member gets an in-app notification: *"You are now the admin of [Group Name]"*

### Add to `groupService.ts`

```ts
// Promote a member to admin
export async function promoteMember(groupId: string, userId: string): Promise<void> {
  const { data: { user: currentUser } } = await supabase.auth.getUser();
  if (!currentUser) throw new Error('Not authenticated');

  // Verify caller is the current admin
  const { data: callerMember } = await supabase
    .from('group_members')
    .select('role')
    .eq('group_id', groupId)
    .eq('user_id', currentUser.id)
    .single();

  if (callerMember?.role !== 'admin') throw new Error('Only admins can promote members.');

  // Promote the target user
  const { error } = await supabase
    .from('group_members')
    .update({ role: 'admin' })
    .eq('group_id', groupId)
    .eq('user_id', userId);

  if (error) throw error;

  // Demote current admin to member
  await supabase
    .from('group_members')
    .update({ role: 'member' })
    .eq('group_id', groupId)
    .eq('user_id', currentUser.id);

  // Notify the newly promoted user
  await supabase.from('notifications').insert({
    user_id: userId,
    type: 'group_promoted',
    title: 'You are now a group admin',
    body: `You have been made the admin of a group.`,
    data: { group_id: groupId },
    read: false,
    created_at: new Date().toISOString(),
  });
}

// Leave a group — enforces promotion rule
export async function leaveGroup(groupId: string): Promise<'deleted' | 'left'> {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) throw new Error('Not authenticated');

  // Check if user is admin
  const { data: myMembership } = await supabase
    .from('group_members')
    .select('role')
    .eq('group_id', groupId)
    .eq('user_id', user.id)
    .single();

  if (myMembership?.role === 'admin') {
    // Count other members
    const { count } = await supabase
      .from('group_members')
      .select('id', { count: 'exact', head: true })
      .eq('group_id', groupId)
      .neq('user_id', user.id);

    if ((count || 0) > 0) {
      // There are other members — caller must promote first
      throw new Error('PROMOTE_REQUIRED');
    }

    // Last member — delete the group
    await supabase.from('groups').delete().eq('id', groupId);
    return 'deleted';
  }

  // Regular member — just leave
  await supabase
    .from('group_members')
    .delete()
    .eq('group_id', groupId)
    .eq('user_id', user.id);

  return 'left';
}
```

```tsx
      Alert.alert('Error', err.message);
    } finally {
      setLoading(false);
    }
  };

  return (
    <View style={{ flex: 1 }}>
      {/* Header */}
      <ThemedText style={Typography.h3}>Choose a New Admin</ThemedText>
      <ThemedText style={{ color: theme.textSecondary }}>
        Select someone to take over "{groupName}" before you leave.
      </ThemedText>

      {/* Member list */}
      <FlatList
        data={members}
        keyExtractor={item => item.user_id}
        renderItem={({ item }) => (
          <Pressable
            onPress={() => setSelected(item.user_id)}
            style={[styles.memberRow, {
              borderColor: selected === item.user_id ? theme.primary : theme.border,

              <Feather name="check-circle" size={20} color={theme.primary} />
            )}
          </Pressable>
        )}
      />

      {/* Confirm button */}
      <Pressable
        style={[styles.confirmBtn, { backgroundColor: theme.primary, opacity: !selected || loading ? 0.5 : 1 }]}
        onPress={handleConfirm}
        disabled={!selected || loading}
      >
        {loading
          ? <ActivityIndicator color="#fff" />
```

## Logic 2 — Removing Members

### Rules
- Admin taps a member's name inside the group → "Remove from group" option appears
- A confirmation alert fires before removing
- Removed user gets an in-app notification: *"You were removed from [Group Name]"*
- Removed user cannot rejoin via invite code (code is still valid for others)
- Admin cannot remove themselves — they must use Leave Group instead

### Add to `groupService.ts`

```ts
// Remove a member from a group (admin only)
export async function removeMember(groupId: string, targetUserId: string): Promise<void> {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) throw new Error('Not authenticated');

  if (targetUserId === user.id) throw new Error('Use Leave Group to remove yourself.');

  // Verify caller is admin
  const { data: callerMember } = await supabase
    .from('group_members')
    .select('role')
    .eq('group_id', groupId)
    .eq('user_id', user.id)
    .single();

  if (callerMember?.role !== 'admin') throw new Error('Only admins can remove members.');

  // Remove the member
  const { error } = await supabase
    .from('group_members')
    .delete()
    .eq('group_id', groupId)
    .eq('user_id', targetUserId);

  if (error) throw error;

  // Fetch group name for notification
  const { data: group } = await supabase
    .from('groups')
    .select('name')
    .eq('id', groupId)
    .single();

  // Notify removed user
  await supabase.from('notifications').insert({
    user_id: targetUserId,
    type: 'group_removed',
    title: 'Removed from group',
    body: `You were removed from "${group?.name || 'a group'}".`,
    data: { group_id: groupId },
    read: false,
    created_at: new Date().toISOString(),
  });
}
```

## Logic 3 — Linked Listing Marked as Rented

### Rules
- When a host marks a listing as rented, any groups linked to that listing have their pinned card updated with a **"Rented"** badge
- The group remains fully active — messages, invites, and all features continue normally
- The host has no ability to close or archive the group
- Members are NOT automatically notified (they'll see the badge next time they open the group)

### Update `listingService.ts` — markAsRented

Find the existing `markPropertyAsRented` (or equivalent) function and add this after updating the listing status:

```ts
// After marking the listing as rented, flag all linked groups
export async function markListingAsRented(listingId: string): Promise<void> {
  const { error } = await supabase
    .from('listings')
    .update({ status: 'rented', rented_at: new Date().toISOString() })
    .eq('id', listingId);

  if (error) throw error;

  // Note: linked groups stay active — no archiving, no deletion.
  // The pinned card UI reads listing.status and shows the badge automatically.
}
```

## Logic 4 — Group Discovery (Opt-In)

### Rules
- `discoverable` toggle defaults to **off** — all groups are private by default
- Admin can toggle it on/off in group settings
- When on: renters browsing that listing in ExploreScreen see a **"X people forming a group"** indicator on the listing card
- Tapping it opens a **Request to Join** flow — the renter sends a request, admin approves or declines
- Approved requests add the renter as a member (subject to member limit)
- Declined requests notify the renter: *"Your request to join [Group Name] was declined"*

### Add to `groupService.ts`

```ts
// Toggle group discoverability
export async function setGroupDiscoverable(groupId: string, discoverable: boolean): Promise<void> {
  const { error } = await supabase
    .from('groups')
    .update({ discoverable })
    .eq('id', groupId);
  if (error) throw error;
}

// Get discoverable groups for a listing (shown on listing card)
export async function getDiscoverableGroupsForListing(listingId: string): Promise<{
  id: string;
  name: string;
  memberCount: number;
}[]> {
  const { data, error } = await supabase
    .from('groups')
    .select('id, name, group_members ( count )')
    .eq('listing_id', listingId)
    .eq('discoverable', true);

  if (error) return [];

  return (data || []).map((g: any) => ({
    id: g.id,
    name: g.name,
    memberCount: g.group_members?.[0]?.count || 0,
  }));
}

// Send a join request
export async function requestToJoinGroup(groupId: string): Promise<void> {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) throw new Error('Not authenticated');

  // Check not already a member
  const { data: existing } = await supabase
    .from('group_members')
    .select('id')
    .eq('group_id', groupId)
    .eq('user_id', user.id)
    .single();

  if (existing) throw new Error('You are already in this group.');

  const { error } = await supabase.from('group_join_requests').insert({
    group_id: groupId,
    user_id: user.id,
    status: 'pending',
  });
  if (error && error.code !== '23505') throw error; // ignore duplicate
}

// Get pending join requests for a group (admin view)
export async function getJoinRequests(groupId: string): Promise<{
  id: string;
  userId: string;
  name: string;
  photo?: string;
  requestedAt: string;
}[]> {
  const { data, error } = await supabase
    .from('group_join_requests')
    .select('id, user_id, created_at, users ( name, photos )')
    .eq('group_id', groupId)
    .eq('status', 'pending')
    .order('created_at', { ascending: true });

  if (error) return [];

  return (data || []).map((r: any) => ({
    id: r.id,
    userId: r.user_id,
    name: r.users?.name || 'Unknown',
    photo: r.users?.photos?.[0],
    requestedAt: r.created_at,
  }));
}

// Approve or decline a join request (admin only)
export async function respondToJoinRequest(
  requestId: string,
  groupId: string,
  response: 'approved' | 'declined'
): Promise<void> {
  const { data: request } = await supabase
    .from('group_join_requests')
    .update({ status: response })
    .eq('id', requestId)
    .select()
    .single();

  if (!request) throw new Error('Request not found.');

  if (response === 'approved') {
    // Check member limit before approving
    const { data: group } = await supabase
      .from('groups')
      .select('created_by, listing_id, listings ( bedrooms )')
      .eq('id', groupId)
      .single();

    const { count } = await supabase
      .from('group_members')
      .select('id', { count: 'exact', head: true })
      .eq('group_id', groupId);

    const creatorPlan = await getUserPlan(group?.created_by);
    const linkedBedrooms = (group as any)?.listings?.bedrooms ?? null;
    const limit = getMemberLimit(creatorPlan, linkedBedrooms);

    if ((count || 0) >= limit) {
      // Auto-decline — group is full
      await supabase
        .from('group_join_requests')
        .update({ status: 'declined' })
        .eq('id', requestId);

      await supabase.from('notifications').insert({
        user_id: request.user_id,
        type: 'join_request_declined',
        title: 'Group is full',
        body: `The group you requested to join is at its member limit.`,
        data: { group_id: groupId },
        read: false,
        created_at: new Date().toISOString(),
      });
      return;
    }

    // Add to group
    await supabase.from('group_members').insert({
      group_id: groupId,
      user_id: request.user_id,
      role: 'member',
    });

    // Notify approved
    const { data: group2 } = await supabase
      .from('groups').select('name').eq('id', groupId).single();

    await supabase.from('notifications').insert({
      user_id: request.user_id,
      type: 'join_request_approved',
      title: 'Join request approved',
      body: `You have been added to "${group2?.name}".`,
      data: { group_id: groupId },
      read: false,
      created_at: new Date().toISOString(),
    });
  } else {
    // Notify declined
    const { data: group2 } = await supabase
      .from('groups').select('name').eq('id', groupId).single();

    await supabase.from('notifications').insert({
      user_id: request.user_id,
      type: 'join_request_declined',
      title: 'Join request declined',
      body: `Your request to join "${group2?.name}" was declined.`,
      data: { group_id: groupId },
      read: false,
      created_at: new Date().toISOString(),
    });
  }
}
```

## Summary

| Rule | Behavior |
|------|----------|
| Admin tries to leave with other members | Blocked — must promote someone first via PromoteAdminScreen |
| Admin is the last member and leaves | Group is permanently deleted |
| Promoted member | Gets in-app notification: "You are now the admin" |
| Admin removes a member | Member is removed + gets in-app notification: "You were removed from [group]" |
| Admin tries to remove themselves | Blocked — directed to Leave Group instead |
| Listing linked to group is marked as rented | Pinned card shows red "RENTED" badge — group stays active |
| Host tries to close linked group | Not possible — host has no group management controls |
| Group discoverable = off (default) | Invisible to non-members — invite only |
| Group discoverable = on | "X people forming a group" appears on listing card in ExploreScreen |
| Renter requests to join | Admin sees request in group — approves or declines |
| Request approved | Renter added as member + notified |
| Request declined or group full | Renter notified with reason |
