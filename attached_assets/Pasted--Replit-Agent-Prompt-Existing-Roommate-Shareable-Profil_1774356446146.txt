# Replit Agent Prompt — Existing Roommate Shareable Profile Link

## Context
Rhome is a React Native / Expo (TypeScript) roommate matching app with a Supabase backend. When a host creates a listing and marks that existing roommates already live in the unit, those roommates are not Rhome users. This feature lets the host share a simple web link with their existing roommate(s). The roommate fills out a short lifestyle preferences form in a browser (no app needed). Their preferences are stored and the AI uses them when scoring compatibility — so new renters are matched against everyone already living there, not just the host.

This builds on Migration 023 (`rooms_available`) which must already be applied.

## What To Build

### 1. Database Migration — `supabase/migrations/024_existing_roommates.sql`

```sql
-- Existing roommates table
CREATE TABLE IF NOT EXISTS existing_roommates (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  listing_id UUID NOT NULL REFERENCES listings(id) ON DELETE CASCADE,
  invite_token TEXT NOT NULL UNIQUE DEFAULT encode(gen_random_bytes(16), 'hex'),
  first_name TEXT,
  age INTEGER,
  occupation TEXT,
  sleep_schedule TEXT CHECK (sleep_schedule IN ('early_bird','night_owl','flexible')),
  cleanliness INTEGER CHECK (cleanliness BETWEEN 1 AND 5),
  smoking BOOLEAN DEFAULT FALSE,
  pets BOOLEAN DEFAULT FALSE,
  lifestyle_tags TEXT[] DEFAULT '{}',
  guests_frequency TEXT CHECK (guests_frequency IN ('rarely','sometimes','often')),
  noise_level TEXT CHECK (noise_level IN ('quiet','moderate','lively')),
  profile_completed BOOLEAN DEFAULT FALSE,
  completed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Hosts can read/manage their existing roommate invites
ALTER TABLE existing_roommates ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Host manages their existing roommates" ON existing_roommates
  FOR ALL USING (
    EXISTS (
      SELECT 1 FROM listings
      WHERE listings.id = existing_roommates.listing_id
        AND listings.host_id = auth.uid()
    )
  );

-- Public read by token — for the web form (no auth required)
CREATE POLICY "Public read by invite token" ON existing_roommates
  FOR SELECT USING (true); -- token-gated at application level

CREATE POLICY "Public update by invite token" ON existing_roommates
  FOR UPDATE USING (true); -- token-gated at application level

-- Index for fast token lookup
CREATE INDEX IF NOT EXISTS idx_existing_roommates_token ON existing_roommates(invite_token);
CREATE INDEX IF NOT EXISTS idx_existing_roommates_listing ON existing_roommates(listing_id);
```

Run this migration in the Supabase SQL Editor.

### 2. Supabase Edge Function — `supabase/functions/submit-roommate-profile/index.ts`

Public Edge Function (no auth required) that lets an existing roommate submit their preferences via their unique token.

```typescript
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

serve(async (req) => {
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders })
  }

  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
  )

  const url = new URL(req.url)
  const token = url.searchParams.get('token')

  // GET — fetch existing roommate record by token (for pre-filling form)
  if (req.method === 'GET') {
    if (!token) {
      return new Response(JSON.stringify({ error: 'Token required' }), {
        status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
      })
    }

    const { data, error } = await supabase
      .from('existing_roommates')
      .select(`
        id, first_name, sleep_schedule, cleanliness, smoking, pets,
        lifestyle_tags, guests_frequency, noise_level, profile_completed,
        listings!existing_roommates_listing_id_fkey (
          address, neighborhood, bedrooms
        )
      `)
      .eq('invite_token', token)
      .single()

    if (error || !data) {
      return new Response(JSON.stringify({ error: 'Invalid or expired link' }), {
        status: 404, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
      })
    }

    return new Response(JSON.stringify(data), {
      headers: { ...corsHeaders, 'Content-Type': 'application/json' }
    })
  }

  // POST — submit roommate preferences
  if (req.method === 'POST') {
    const body = await req.json()
    const {
      token: bodyToken,
      firstName,
      sleepSchedule,
      cleanliness,
      smoking,
      pets,
      lifestyleTags,
      guestsFrequency,
      noiseLevel
    } = body

    const useToken = bodyToken || token

    if (!useToken) {
      return new Response(JSON.stringify({ error: 'Token required' }), {
        status: 400, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
      })
    }

    // Validate the token exists
    const { data: existing } = await supabase
      .from('existing_roommates')
      .select('id, profile_completed')
      .eq('invite_token', useToken)
      .single()

    if (!existing) {
      return new Response(JSON.stringify({ error: 'Invalid or expired link' }), {
        status: 404, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
      })
    }

    // Update the record
    const { error: updateError } = await supabase
      .from('existing_roommates')
      .update({
        first_name: firstName,
        sleep_schedule: sleepSchedule,
        cleanliness: parseInt(cleanliness) || null,
        smoking: Boolean(smoking),
        pets: Boolean(pets),
        lifestyle_tags: lifestyleTags || [],
        guests_frequency: guestsFrequency,
        noise_level: noiseLevel,
        profile_completed: true,
        completed_at: new Date().toISOString()
      })
      .eq('invite_token', useToken)

    if (updateError) {
      return new Response(JSON.stringify({ error: updateError.message }), {
        status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' }
      })
    }

    return new Response(JSON.stringify({ success: true }), {
      headers: { ...corsHeaders, 'Content-Type': 'application/json' }
    })
  }

  return new Response('Method not allowed', { status: 405, headers: corsHeaders })
})
```

Deploy: `supabase functions deploy submit-roommate-profile --no-verify-jwt`

The `--no-verify-jwt` flag is required because this endpoint is accessed without authentication.

### 4. Service — `src/services/existingRoommateService.ts`

```typescript
import { supabase } from './supabase'
import * as Sharing from 'expo-sharing'

const SUPABASE_PROJECT_URL = process.env.EXPO_PUBLIC_SUPABASE_URL || ''

export interface ExistingRoommateRecord {
  id: string
  listingId: string
  inviteToken: string
  firstName?: string
  sleepSchedule?: string
  cleanliness?: number
  smoking?: boolean
  pets?: boolean
  lifestyleTags?: string[]
  guestsFrequency?: string
  noiseLevel?: string
  profileCompleted: boolean
  completedAt?: string
}

// Create invite records for a listing (call after listing is created)
export async function createExistingRoommateInvites(
  listingId: string,
  count: number
): Promise<ExistingRoommateRecord[]> {
  // Create one record per existing roommate
  const records = Array.from({ length: count }, () => ({ listing_id: listingId }))
  const { data, error } = await supabase
    .from('existing_roommates')
    .insert(records)
    .select()

  if (error) throw error
  return (data || []).map(mapRecord)
}

// Get all existing roommates for a listing
export async function getExistingRoommates(listingId: string): Promise<ExistingRoommateRecord[]> {
  const { data, error } = await supabase
    .from('existing_roommates')
    .select('*')
    .eq('listing_id', listingId)
    .order('created_at', { ascending: true })

  if (error) throw error
  return (data || []).map(mapRecord)
}

// Generate the shareable link for a roommate token
export function getRoommateInviteLink(token: string): string {
  // Points to the hosted web form
  return `${SUPABASE_PROJECT_URL}/storage/v1/object/public/web/roommate-profile.html?token=${token}`
}

// Share the link via the device share sheet
export async function shareRoommateInviteLink(token: string, roommateNumber?: number): Promise<void> {
  const link = getRoommateInviteLink(token)
  const message = roommateNumber
    ? `Hey! I'm looking for a new roommate for our place and I'm using Rhome to help find the right fit. Can you fill out this quick form so the app can match us with someone compatible? Takes 2 minutes: ${link}`
    : `Hey! I'm using Rhome to find a new roommate. Fill out this quick form so we can find someone we'll all get along with: ${link}`

  await Sharing.shareAsync(link, {
    dialogTitle: 'Share with your roommate',
    UTI: 'public.url',
    mimeType: 'text/plain'
  })
}

// Check completion status for a listing's existing roommates
export async function getRoommateCompletionStatus(listingId: string): Promise<{
  total: number
  completed: number
  allComplete: boolean
}> {
  const roommates = await getExistingRoommates(listingId)
  const completed = roommates.filter(r => r.profileCompleted).length
  return {
    total: roommates.length,
    completed,
    allComplete: roommates.length > 0 && completed === roommates.length
  }
}

function mapRecord(r: any): ExistingRoommateRecord {
  return {
    id: r.id,
    listingId: r.listing_id,
    inviteToken: r.invite_token,
    firstName: r.first_name,
    sleepSchedule: r.sleep_schedule,
    cleanliness: r.cleanliness,
    smoking: r.smoking,
    pets: r.pets,
    lifestyleTags: r.lifestyle_tags,
    guestsFrequency: r.guests_frequency,
    noiseLevel: r.noise_level,
    profileCompleted: r.profile_completed,
    completedAt: r.completed_at
```

### 6. Update AI Scoring to Include Existing Roommates

In ALL AI compatibility scoring functions (evaluateGroupForHost, getTopRentersForListing, calculate-match-scores Edge Function, and any function that scores renter vs listing), update to include existing roommates in the score:

**Update compatibility scoring utility** — add to `src/utils/compatibilityScoring.ts` (or wherever scoring logic lives):

```typescript
import { supabase } from '../services/supabase'

// Fetch existing roommates for a listing (completed profiles only)
export async function getExistingRoommatesForListing(listingId: string) {
  const { data } = await supabase
    .from('existing_roommates')
    .select('*')
    .eq('listing_id', listingId)
    .eq('profile_completed', true)

  return data || []
}

// Score a renter profile against an existing roommate's preferences
export function scoreRenterVsExistingRoommate(
  renterProfile: any,
  existingRoommate: any
): number {
  let score = 100
  const penalties: string[] = []

  // Sleep schedule mismatch (-25 points)
  if (
    existingRoommate.sleep_schedule &&
    renterProfile.sleep_schedule &&
    existingRoommate.sleep_schedule !== 'flexible' &&
    renterProfile.sleep_schedule !== 'flexible' &&
    existingRoommate.sleep_schedule !== renterProfile.sleep_schedule
  ) {
    score -= 25
    penalties.push('Sleep schedule conflict with existing roommate')
  }

  // Cleanliness gap > 2 (-20 points)
  if (existingRoommate.cleanliness && renterProfile.cleanliness) {
    const gap = Math.abs(existingRoommate.cleanliness - renterProfile.cleanliness)
    if (gap >= 3) {
      score -= 20
      penalties.push('Cleanliness standards very different from existing roommate')
    } else if (gap === 2) {
      score -= 10
    }
  }

  // Smoking conflict (-30 points — deal breaker)
  if (existingRoommate.smoking === false && renterProfile.smoking === true) {
    score -= 30
    penalties.push('Existing roommate is non-smoking')
  }

  // Pets conflict (-20 points)
  if (existingRoommate.pets === false && renterProfile.pets === true) {
    score -= 20
    penalties.push('Existing roommate does not allow pets')
  }

  // Guests frequency conflict
  if (existingRoommate.guests_frequency === 'rarely' && renterProfile.lifestyle_tags?.includes('social')) {
    score -= 10
    penalties.push('Existing roommate prefers fewer guests')
  }

  return { score: Math.max(0, score), penalties }
}

// Combined score: renter vs host + renter vs all existing roommates
export function calculateCombinedCompatibility(
  renterProfile: any,
  hostProfile: any,
  existingRoommates: any[],
  baseScore: number // score already calculated vs host
): {
  finalScore: number
  existingRoommateScores: { name: string; score: number; penalties: string[] }[]
  lowestScore: number
} {
  if (existingRoommates.length === 0) {
    return {
      finalScore: baseScore,
      existingRoommateScores: [],
      lowestScore: baseScore
    }
  }

  const roommateScores = existingRoommates.map(er => {
    const { score, penalties } = scoreRenterVsExistingRoommate(renterProfile, er)
    return {
      name: er.first_name || 'Existing roommate',
      score,
      penalties
    }
  })

  // Final score = average of host score + all existing roommate scores
  const allScores = [baseScore, ...roommateScores.map(r => r.score)]
  const finalScore = Math.round(allScores.reduce((a, b) => a + b, 0) / allScores.length)
  const lowestScore = Math.min(...allScores)

  return { finalScore, existingRoommateScores: roommateScores, lowestScore }
}
```
