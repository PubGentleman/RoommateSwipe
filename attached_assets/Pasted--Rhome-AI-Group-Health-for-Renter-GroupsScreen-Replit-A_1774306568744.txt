# Rhome — AI Group Health for Renter GroupsScreen (Replit Agent Prompt)

Paste this into Replit Agent after the other AI prompts are complete.

---

## Context

The renter's GroupsScreen lists all the groups they belong to.
Right now it shows basic info — group name, member count, maybe a listing.
There is no indication of whether a group is actually compatible or likely
to succeed in finding an apartment together.

Renters need to know:
- Which of my groups has the best shot at finding an apartment?
- Does my group have any conflicts I should know about before we start looking?
- Is one group stalled because budgets don't overlap?

This prompt adds:
1. A group health score on every group card in the list
2. Conflict warning badges for groups with serious problems
3. A pinned "Best Group" banner at the top — which group should they focus on
4. Inside each group, a full compatibility breakdown between all members

### PART 1: Group Health Score utility

Create: utils/groupHealthScore.ts

  import { supabase } from '../lib/supabase';
  import { detectGroupConflicts, calculateGroupCompatibilityScore } from './transitMatching';

  export type GroupHealthStatus = 'strong' | 'good' | 'at_risk' | 'conflict';

  export interface GroupHealthResult {
    score: number;                  // 0–100
    status: GroupHealthStatus;
    statusLabel: string;            // "Strong", "Good", "Needs Work", "Conflict"
    statusColor: string;            // hex color
    topConflict: string | null;     // single most important conflict to surface
    conflicts: string[];            // all conflicts
    suggestions: string[];          // how to fix
    sharedNeighborhoods: string[];  // neighborhoods that work for everyone
    memberCount: number;
    desiredBedrooms: number | null;
    readyToSearch: boolean;         // true if group has prefs + no hard conflicts
  }

  export async function getGroupHealth(
    groupId: string,
    currentUserId: string
  ): Promise<GroupHealthResult | null> {
    // Fetch group and members
    const { data: group } = await supabase
      .from('groups')
      .select('*, group_members(user_id)')
      .eq('id', groupId)
      .single();

    if (!group) return null;

    const memberIds: string[] = (group.group_members ?? []).map((m: any) => m.user_id);
    if (memberIds.length < 2) {
      return {
        score: 0,
        status: 'at_risk',
        statusLabel: 'Needs More Members',
        statusColor: '#f39c12',
        topConflict: 'Your group only has 1 member. Invite someone to get started.',
        conflicts: [],
        suggestions: ['Invite a roommate to join your group'],
        sharedNeighborhoods: [],
        memberCount: memberIds.length,
        desiredBedrooms: null,
        readyToSearch: false,
      };
    }

    // Fetch member profiles with apartment prefs
    const { data: members } = await supabase
      .from('users')
      .select(`
        id,
        full_name,
        profile:profiles(
          cleanliness,
          sleep_schedule,
          smoking,
          pets,
          move_in_date,
          budget_max,
          preferred_trains,
          budget_per_person_min,
          budget_per_person_max,
          desired_bedrooms,
          location_flexible,
          wfh,
          amenity_must_haves,
          apartment_prefs_complete
        )
      `)
      .in('id', memberIds);

    if (!members || members.length === 0) return null;

    // Map to RoommateProfile shape for compatibility functions
    const profiles = members.map(m => ({
      id: m.id,
      name: m.full_name,
      lifestyle: {
        cleanliness: m.profile?.cleanliness,
        workSchedule: m.profile?.sleep_schedule,
        smoking: m.profile?.smoking,
        pets: m.profile?.pets,
      },
      budget: m.profile?.budget_max,
      preferences: { moveInDate: m.profile?.move_in_date },
      apartmentPrefs: m.profile?.apartment_prefs_complete ? {
        desiredBedrooms: m.profile.desired_bedrooms,
        budgetPerPersonMin: m.profile.budget_per_person_min,
        budgetPerPersonMax: m.profile.budget_per_person_max,
        preferredTrains: m.profile.preferred_trains ?? [],
        amenityMustHaves: m.profile.amenity_must_haves ?? [],
        moveInDate: m.profile.move_in_date,
        locationFlexible: m.profile.location_flexible,
        wfh: m.profile.wfh,
        apartmentPrefsComplete: true,
      } : undefined,
    }));

    // Run conflict detection
    const conflictResult = detectGroupConflicts(profiles as any);

    // Calculate pairwise compatibility scores
    const pairs: Array<[any, any]> = [];
    for (let i = 0; i < profiles.length; i++) {
      for (let j = i + 1; j < profiles.length; j++) {
        pairs.push([profiles[i], profiles[j]]);
      }
    }

    const pairScores = pairs.map(([a, b]) =>
      calculateGroupCompatibilityScore(a as any, b as any).total
    );
    const avgScore = pairScores.length > 0
      ? Math.round(pairScores.reduce((s, n) => s + n, 0) / pairScores.length)
      : 50;

    // How many members have completed apartment prefs?
    const prefsComplete = profiles.filter(p => p.apartmentPrefs?.apartmentPrefsComplete).length;
    const allPrefsComplete = prefsComplete === profiles.length;

    // Desired bedrooms consensus
    const bedroomVotes = profiles
      .map(p => p.apartmentPrefs?.desiredBedrooms)
      .filter(Boolean) as number[];
    const desiredBedrooms = bedroomVotes.length > 0
      ? Math.round(bedroomVotes.reduce((s, n) => s + n, 0) / bedroomVotes.length)
      : null;

    // Penalize score if prefs are incomplete
    const prefsBonus = allPrefsComplete ? 0 : -15;
    const finalScore = Math.max(0, Math.min(100, avgScore + prefsBonus));

    // Status
    let status: GroupHealthStatus;
    let statusLabel: string;
    let statusColor: string;

    if (conflictResult.conflicts.length >= 2 || finalScore < 40) {
      status = 'conflict';
      statusLabel = 'Conflict';
      statusColor = '#e74c3c';
    } else if (conflictResult.conflicts.length === 1 || finalScore < 65) {
      status = 'at_risk';
      statusLabel = 'Needs Work';
      statusColor = '#f39c12';
    } else if (finalScore < 80) {
      status = 'good';
      statusLabel = 'Good';
      statusColor = '#3498db';
    } else {
      status = 'strong';
      statusLabel = 'Strong';
      statusColor = '#2ecc71';
    }

    // Top conflict to surface
    const topConflict = conflictResult.conflicts.length > 0
      ? conflictResult.conflicts[0]
      : !allPrefsComplete
        ? `${profiles.length - prefsComplete} member(s) haven't set apartment preferences yet`
        : null;

    const readyToSearch =
      allPrefsComplete &&
      conflictResult.conflicts.length === 0 &&
      finalScore >= 60;

    return {
      score: finalScore,
      status,
      statusLabel,
      statusColor,
      topConflict,
      conflicts: conflictResult.conflicts,
      suggestions: conflictResult.suggestions,
      sharedNeighborhoods: conflictResult.sharedNeighborhoods,
      memberCount: members.length,
      desiredBedrooms,
      readyToSearch,
    };
  }

  // Get health for multiple groups at once
  export async function getGroupsHealth(
    groupIds: string[],
    currentUserId: string
  ): Promise<Record<string, GroupHealthResult>> {
    const results: Record<string, GroupHealthResult> = {};
    await Promise.all(
      groupIds.map(async (id) => {
        const result = await getGroupHealth(id, currentUserId);
        if (result) results[id] = result;
      })
    );
    return results;
  }

  import { getGroupsHealth, GroupHealthResult } from '../../utils/groupHealthScore';

  const [groupHealthScores, setGroupHealthScores] = useState<Record<string, GroupHealthResult>>({});
  const [bestGroupId, setBestGroupId] = useState<string | null>(null);

  // Load health scores after groups load
  useEffect(() => {
    if (groups.length > 0 && user) {
      const ids = groups.map((g: any) => g.id);
      getGroupsHealth(ids, user.id).then(scores => {
        setGroupHealthScores(scores);

        // Find the best group (highest score that is ready to search)
        const ranked = Object.entries(scores)
          .filter(([, h]) => h.score > 0)
          .sort(([, a], [, b]) => b.score - a.score);
        if (ranked.length > 0) setBestGroupId(ranked[0][0]);
      });
    }
  }, [groups]);
