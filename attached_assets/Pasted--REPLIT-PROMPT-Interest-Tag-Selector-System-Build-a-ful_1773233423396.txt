**PART 6 — WIRE INTO MATCHING ALGORITHM**

In `utils/matchingAlgorithm.ts`, add interest tag overlap scoring to the `calculateCompatibility()` function:

```ts
const calculateInterestScore = (
  user1Tags: string[],
  user2Tags: string[]
): number => {
  if (!user1Tags?.length || !user2Tags?.length) return 0;
  const set1 = new Set(user1Tags);
  const overlap = user2Tags.filter(tag => set1.has(tag)).length;
  const maxPossible = Math.min(user1Tags.length, user2Tags.length);
  return maxPossible > 0 ? (overlap / maxPossible) * 100 : 0;
};
```

Add this to the overall compatibility score with a **20% weighting**:

```ts
const interestScore = calculateInterestScore(
  user1.profile?.interests ?? [],
  user2.profile?.interests ?? []
);

// Add to total score
totalScore += interestScore * 0.20;
```

Also add specific tag compatibility bonuses — if both users share tags that are directly conflicting when mismatched, apply a penalty instead of a bonus:

```ts
const CONFLICTING_PAIRS = [
  ['smoker', 'non_smoker'],
  ['pet_friendly', 'no_pets'],
  ['early_bird', 'night_owl'],
  ['host_gatherings', 'rarely_guests'],
];

CONFLICTING_PAIRS.forEach(([tag1, tag2]) => {
  const u1HasTag1 = user1Tags.includes(tag1);
  const u1HasTag2 = user1Tags.includes(tag2);
  const u2HasTag1 = user2Tags.includes(tag1);
  const u2HasTag2 = user2Tags.includes(tag2);

  // If one user is a smoker and the other is non-smoker, apply -15 penalty
  if ((u1HasTag1 && u2HasTag2) || (u1HasTag2 && u2HasTag1)) {
    totalScore -= 15;
  }
});
```
