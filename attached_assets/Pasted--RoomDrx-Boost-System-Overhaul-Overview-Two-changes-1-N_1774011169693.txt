# RoomDrx — Boost System Overhaul

## Overview

Two changes:
1. **New unified boost tiers** — 6h/$2.99 · 12h/$4.99 (most popular) · 24h/$7.99, applied to both renter profile boosts and host listing boosts
2. **Rotation system** — boosted items shuffle order every 30 minutes so no single boost permanently holds position #1

## Part 2 — Boost Rotation System

### How it works

Every 30 minutes, the order of boosted listings/profiles is shuffled using a **deterministic seed** based on:
- Current 30-minute time slot
- The viewing user's ID

This means:
- Every user sees a different random order of boosted items (fairness)
- The order is stable within each 30-minute window (no jitter on re-render)
- Every boosted listing gets equal expected exposure over time

### 2a. New shared utility — `utils/boostRotation.ts` (create new file)

```ts
/**
 * Deterministic rotation for boosted listings/profiles.
 * Boosted items shuffle order every ROTATION_WINDOW_MS using a seeded shuffle,
 * so no single boost permanently occupies position #1.
 */

const ROTATION_WINDOW_MS = 30 * 60 * 1000; // 30 minutes

/** Simple 32-bit hash of a string */
function hashString(str: string): number {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = Math.imul(31, hash) + str.charCodeAt(i) | 0;
  }
  return hash >>> 0;
}

/** Current rotation slot index (changes every ROTATION_WINDOW_MS) */
function currentSlot(): number {
  return Math.floor(Date.now() / ROTATION_WINDOW_MS);
}

/** LCG-based deterministic shuffle — does not mutate input array */
function seededShuffle<T>(arr: T[], seed: number): T[] {
  const a = [...arr];
  let s = seed;
  for (let i = a.length - 1; i > 0; i--) {
    // LCG step
    s = (Math.imul(1664525, s) + 1013904223) | 0;
    const j = (s >>> 0) % (i + 1);
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

/**
 * Returns a rotated copy of boosted items.
 * Order changes every 30 minutes and is unique per viewerId.
 * Non-boosted items are returned as-is after boosted.
 */
export function applyBoostRotation<T>(
  boosted: T[],
  normal: T[],
  viewerId: string
): T[] {
  if (boosted.length <= 1) return [...boosted, ...normal];
  const seed = currentSlot() ^ hashString(viewerId);
  const rotated = seededShuffle(boosted, seed);
  return [...rotated, ...normal];
}

/**
 * For use in sort comparisons — returns a stable rotation index
 * for a boosted item so sort() produces the rotated order.
 */
export function getBoostRotationIndex(
  itemId: string,
  allBoostedIds: string[],
  viewerId: string
): number {
  const seed = currentSlot() ^ hashString(viewerId);
  const shuffled = seededShuffle(allBoostedIds, seed);
  return shuffled.indexOf(itemId);
}
```
