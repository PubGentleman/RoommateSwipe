**8. Supabase migration (add to a new migration file)**

```sql
-- Add visible_to_hosts to groups table
ALTER TABLE groups ADD COLUMN IF NOT EXISTS visible_to_hosts BOOLEAN DEFAULT true;
ALTER TABLE groups ADD COLUMN IF NOT EXISTS host_initiated BOOLEAN DEFAULT false;

-- RLS: hosts can read roommate groups where visible_to_hosts = true
CREATE POLICY "hosts_can_view_renter_groups" ON groups
  FOR SELECT
  USING (
    type = 'roommate'
    AND visible_to_hosts = true
    AND auth.uid() IN (SELECT id FROM users WHERE is_host = true)
  );
```
