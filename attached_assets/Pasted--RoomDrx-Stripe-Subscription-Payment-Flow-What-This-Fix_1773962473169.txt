# RoomDrx — Stripe Subscription Payment Flow

## What This Fixes
Currently `upgradeToPlus()` and `upgradeToElite()` in `AuthContext.tsx` skip Stripe entirely — they just write the new plan directly to Supabase. This means any user can upgrade for free. This prompt wires the Stripe PaymentSheet into the upgrade flow so a real card is charged before the plan activates.

## Step 1 — Update the `create-subscription` Edge Function

Open `supabase/functions/create-subscription/index.ts` and make sure it looks like this:

```ts
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts';
import Stripe from 'https://esm.sh/stripe@14.21.0?target=deno';
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const stripe = new Stripe(Deno.env.get('STRIPE_SECRET_KEY')!, { apiVersion: '2023-10-16' });
const supabase = createClient(
  Deno.env.get('SUPABASE_URL')!,
  Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
);

serve(async (req) => {
  const { userId, priceId, email } = await req.json();

  // Get or create Stripe customer
  let customerId: string;
  const { data: sub } = await supabase
    .from('subscriptions')
    .select('stripe_customer_id')
    .eq('user_id', userId)
    .single();

  if (sub?.stripe_customer_id) {
    customerId = sub.stripe_customer_id;
  } else {
    const customer = await stripe.customers.create({ email, metadata: { supabase_user_id: userId } });
    customerId = customer.id;
    await supabase.from('subscriptions').update({ stripe_customer_id: customerId }).eq('user_id', userId);
  }

  // Create PaymentIntent with subscription setup
  const paymentIntent = await stripe.paymentIntents.create({
    amount: 0, // will be set by subscription
    currency: 'usd',
    customer: customerId,
    setup_future_usage: 'off_session',
    metadata: { userId, priceId },
  });

  // Create subscription with payment_behavior = default_incomplete
  const subscription = await stripe.subscriptions.create({
    customer: customerId,
    items: [{ price: priceId }],
    payment_behavior: 'default_incomplete',
    payment_settings: { save_default_payment_method: 'on_subscription' },
    expand: ['latest_invoice.payment_intent'],
  });

  const invoice = subscription.latest_invoice as any;
  const clientSecret = invoice?.payment_intent?.client_secret;

  return new Response(JSON.stringify({ clientSecret, subscriptionId: subscription.id }), {
    headers: { 'Content-Type': 'application/json' },
  });
});
```

Deploy this with:
```bash
supabase functions deploy create-subscription
```

### Add the shared `chargeAndUpgrade` helper

Add this function inside the `AuthProvider` component, before `upgradeToPlus`:

```ts
const chargeAndUpgrade = async (plan: string, billingCycle: 'monthly' | '3month' | 'annual') => {
  if (!user) return;

  const priceId = getPriceId(plan, billingCycle);
  if (!priceId) {
    Alert.alert('Error', 'Could not find pricing for this plan. Please try again.');
    return;
  }

  try {
    // Call edge function to get client secret
    const { data, error } = await supabase.functions.invoke('create-subscription', {
      body: { userId: user.id, priceId, email: user.email },
    });

    if (error || !data?.clientSecret) {
      throw new Error(error?.message || 'Failed to initialize payment');
    }

    // Present Stripe PaymentSheet
    const { initPaymentSheet, presentPaymentSheet } = useStripe();

    const { error: initError } = await initPaymentSheet({
      paymentIntentClientSecret: data.clientSecret,
      merchantDisplayName: 'RoomDrx',
      allowsDelayedPaymentMethods: false,
    });

    if (initError) throw new Error(initError.message);

    const { error: paymentError } = await presentPaymentSheet();

    if (paymentError) {
      if (paymentError.code !== 'Canceled') {
        Alert.alert('Payment Failed', paymentError.message);
      }
      return; // User canceled or payment failed — don't upgrade
    }

    // Payment succeeded — now activate the plan in Supabase
    const expiresAt = getExpiryForCycle(billingCycle);
    const amount = getBillingAmount(plan as any, billingCycle);
    const prevHistory = user.subscription?.billingHistory || [];

    const planKey = plan.startsWith('host_') ? plan.replace('host_', '') : plan;

    const updatedUser: User = {
      ...user,
      subscription: {
        plan: planKey as any,
        status: 'active',
        expiresAt,
        billingCycle,
        billingHistory: [
          { date: new Date().toISOString(), amount, description: `${planKey} - ${billingCycle}` },
          ...prevHistory,
        ].slice(0, 10),
      },
    };

    await StorageService.setCurrentUser(updatedUser);
    await StorageService.addOrUpdateUser(updatedUser);
    await supabase
      .from('subscriptions')
      .update({
        plan: planKey,
        billing_cycle: billingCycle,
        status: 'active',
        current_period_end: expiresAt.toISOString(),
        cancel_at_period_end: false,
        stripe_subscription_id: data.subscriptionId,
      })
      .eq('user_id', user.id);

    setUser(updatedUser);
    Alert.alert('Success!', `You're now on the ${planKey.charAt(0).toUpperCase() + planKey.slice(1)} plan.`);

  } catch (err: any) {
    Alert.alert('Error', err.message || 'Something went wrong. Please try again.');
  }
};
```

> **Note:** `useStripe()` is a React hook and must be called inside a component, not inside a context function. See the alternative approach below if you hit a "hooks can't be called here" error.

## Alternative: Move Payment Logic to `PlanSelectionScreen`

If you run into issues calling `useStripe()` inside `AuthContext`, keep the payment logic in the screen instead. In `PlanSelectionScreen.tsx`, update `handleConfirmPlan`:

```ts
import { useStripe } from '@stripe/stripe-react-native';
import { getPriceId } from '../../constants/stripePrices';

// Inside the component:
const { initPaymentSheet, presentPaymentSheet } = useStripe();

const handleConfirmPlan = async (planIdOverride?: string) => {
  const planId = planIdOverride || selectedPlanId;
  if (!planId || !user) return;

  const plan = plans.find(p => p.id === planId);
  if (plan?.isFree) {
    // Free plan — no payment needed
    await completeOnboardingStep('complete');
    return;
  }

  setProcessing(true);

  const priceId = getPriceId(planId, billingCycle);
  if (!priceId) {
    Alert.alert('Error', 'Pricing not found for this plan.');
    setProcessing(false);
    return;
  }

  try {
    const { data, error } = await supabase.functions.invoke('create-subscription', {
      body: { userId: user.id, priceId, email: user.email },
    });

    if (error || !data?.clientSecret) throw new Error('Failed to initialize payment');

    const { error: initError } = await initPaymentSheet({
      paymentIntentClientSecret: data.clientSecret,
      merchantDisplayName: 'RoomDrx',
      allowsDelayedPaymentMethods: false,
    });
    if (initError) throw new Error(initError.message);

    const { error: paymentError } = await presentPaymentSheet();
    if (paymentError) {
      if (paymentError.code !== 'Canceled') Alert.alert('Payment Failed', paymentError.message);
      setProcessing(false);
      setShowConfirm(false);
      return;
    }

    // Payment succeeded — activate plan
    if (isHost) {
      if (planId === 'starter') await upgradeHostPlan('starter', billingCycle);
      else if (planId === 'pro') await upgradeHostPlan('pro', billingCycle);
      else if (planId === 'business') await upgradeHostPlan('business', billingCycle);
    } else {
      if (planId === 'plus') await upgradeToPlus(billingCycle);
      else if (planId === 'elite') await upgradeToElite(billingCycle);
    }

    await completeOnboardingStep('complete');
    Alert.alert('Welcome!', `You're now on the ${planId.charAt(0).toUpperCase() + planId.slice(1)} plan.`);

  } catch (err: any) {
    Alert.alert('Error', err.message || 'Something went wrong.');
  } finally {
    setProcessing(false);
    setShowConfirm(false);
  }
};
```

This approach is cleaner since hooks work naturally in components.

## Result
- Users must enter a real credit card before their plan activates
- Stripe handles payment retry, card decline messaging, and 3D Secure
- Subscription renewals are handled automatically via webhook
- Failed payments downgrade the user back to Basic via the webhook handler
- Free plan selection still works instantly with no payment prompt
